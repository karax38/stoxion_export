*&---------------------------------------------------------------------*
*& Report  ZSTOXION_EXPORT
*&---------------------------------------------------------------------*
*& Zweck   : Template-konformer Datenexport fuer die stoxion
*&           Safety-Stock-Optimierung (Pilot Ansmann AG).
*&           Erzeugt eine oder mehrere XLSX-Arbeitsmappen mit je SAP-Tabelle
*&           einem Reiter und zwei Kopfzeilen (Bezeichnung + technischer Feldname),
*&           identisch zur Reiterstruktur des Excel-Templates
*&           "Soll-Sicherheitsbestand_Template.xlsx".
*&
*& Reiter   : T001, T001W, T006, T134, V438A, V_399D_E,
*&            MARA, MATDOC, MARC, MBEW
*&
*& Version 2 - Aenderungen ggue. v1:
*&   - Mehrere Werke je Lauf (S_WERKS), alle in einem Dateisatz
*&   - Neue Filter: Material, Materialart, Warengruppe, Dispomerkmal,
*&     ABC-Kennzeichen, Sonderbestandskennzeichen (nur Filter, keine
*&     neuen Spalten -> Template bleibt unveraendert)
*&   - Top-N entfernt (Auswahl erfolgt in stoxion)
*&   - S/4-Weiche entfernt: immer MSEG+MKPF. Auf S/4HANA ist MSEG ein
*&     Compatibility View auf MATDOC -> identische Daten, ein Codepfad
*&   - Bewegungsdaten paketiert (Material-Chunks)
*&   - Hintergrundjob-Button auf dem Selektionsbild (erzwingt
*&     Server-Export), Batch-Schutz bei sy-batch
*&   - AUTHORITY-CHECKs: M_MATE_WRK je Werk, S_DATASET vor OPEN DATASET
*&   - MARM-Filter als DB-Join statt Doppel-Loop
*&
*& Version 3 - Aenderungen ggue. v2:
*&   - XLSX-Workbook-Ausgabe statt einzelner TXT-Dateien
*&   - Bewegungsdaten auf gefilterte Material/Werk-Kombinationen begrenzt
*&   - Zahlen/Datum explizit im Parser-Format ausgegeben
*&   - Selektionsbildschirm deutsch/englisch beschriftet
*&
*& Version 4 - Aenderungen ggue. v3:
*&   - Workbook-Reihenfolge und Kopfzeilen an MVP-Template main angeglichen
*&   - Bewegungsarten default auf MVP-relevante Nachfragearten gesetzt
*&   - Bewegungsdaten-Defaultzeitraum auf ca. 24 Monate begrenzt
*&   - Abbruch vor XLSX-Erzeugung, wenn MATDOC das Excel-Zeilenlimit
*&     ueberschreitet
*&
*& Version 5 - Aenderungen ggue. v4:
*&   - MATDOC wird an Material/Werk-Grenzen auf mehrere Arbeitsmappen
*&     verteilt, bevor ein Excel-Reiter das Zeilenlimit ueberschreitet
*&   - Bei mehreren Arbeitsmappen wird automatisch ein ZIP geschrieben
*&   - MARA/MARC/MBEW werden je Arbeitsmappe auf das jeweilige
*&     Material/Werk-Paket eingeschraenkt
*&
*& Release  : Min. NW 7.40 (Inline-Deklarationen).
*&            Lauffaehig auf ECC 6.0 und S/4HANA.
*&
*& Hinweis  : Als lokales Objekt ($TMP) anlegbar -> KEIN Transport noetig.
*&            Empfehlung fuer Voll-Export: Button "Hintergrundjob
*&            einplanen" verwenden; Exportdatei danach aus AL11 abholen
*&            (z.B. via CG3Y).
*&            Datenschutz: Es werden KEINE personenbezogenen Daten
*&            exportiert (kein Benutzername aus MKPF o.ae.).
*&
*& Hard-Filter (per Checkbox, default aktiv):
*&   - Material/Werk-Kombinationen mit Dispobereich (MDMA) entfernt
*&   - Materialien mit alternativen Mengeneinheiten (MARM-Satz mit
*&     MEINH <> Basis-ME) komplett entfernt
*&
*& Formatkontrakt (stoxion-Parser):
*&   - XLSX-Arbeitsmappe(n), ein Reiter je Template-Reiter, 2 Kopfzeilen
*&   - Datum roh als JJJJMMTT, Mengen/Preise mit '.' als Dezimaltrenner,
*&     MATNR mit fuehrenden Nullen
*&   - Zellen als Text geschrieben, damit IDs unveraendert bleiben
*&---------------------------------------------------------------------*

*----------------------------------------------------------------------*
START-OF-SELECTION.
*----------------------------------------------------------------------*

* --- Batch-Schutz: Frontend-Download geht im Hintergrund nicht --------
  IF sy-batch = 'X' AND p_front = 'X'.
    p_front = space.
    p_serv  = 'X'.
    MESSAGE 'Hintergrundlauf: Ausgabe auf Server-Export umgestellt.'
      TYPE 'S'.
  ENDIF.

  PERFORM validate_target_path.

* --- 0) Werke aufloesen, Berechtigungen, Org-Daten --------------------
  SELECT werks, bwkey FROM t001w
    INTO CORRESPONDING FIELDS OF TABLE @gt_plants
    WHERE werks IN @s_werks.
  IF gt_plants IS INITIAL.
    MESSAGE 'Kein Werk zur Selektion gefunden.' TYPE 'E'.
  ENDIF.

  LOOP AT gt_plants ASSIGNING FIELD-SYMBOL(<ls_plant>).
    AUTHORITY-CHECK OBJECT 'M_MATE_WRK'
      ID 'ACTVT' FIELD '03'
      ID 'WERKS' FIELD <ls_plant>-werks.
    IF sy-subrc <> 0.
      MESSAGE |Keine Berechtigung fuer Werk { <ls_plant>-werks }.|
        TYPE 'E'.
    ENDIF.

    SELECT SINGLE bukrs FROM t001k
      INTO @<ls_plant>-bukrs
      WHERE bwkey = @<ls_plant>-bwkey.
    IF sy-subrc <> 0.
      MESSAGE |Kein Buchungskreis zu Werk { <ls_plant>-werks }.| TYPE 'E'.
    ENDIF.
    SELECT SINGLE waers FROM t001
      INTO @<ls_plant>-waers
      WHERE bukrs = @<ls_plant>-bukrs.
    IF sy-subrc <> 0.
      MESSAGE |Buchungskreis { <ls_plant>-bukrs } nicht gefunden.|
        TYPE 'E'.
    ENDIF.
  ENDLOOP.
  SORT gt_plants BY werks.

* --- 1) Kandidaten: MARC + MARA mit allen Filtern in einem Join -------
  SELECT c~matnr, c~werks, c~lvorm, c~dismm, c~eisbe, c~plifz,
         c~webaz, c~pstat, c~maabc, c~wzeit,
         a~mtart, a~meins
    FROM marc AS c
    INNER JOIN mara AS a ON a~matnr = c~matnr
    INTO CORRESPONDING FIELDS OF TABLE @gt_marc
    WHERE c~werks IN @s_werks
      AND c~matnr IN @s_matnr
      AND c~lvorm =  @space
      AND c~dismm IN @s_dismm
      AND c~maabc IN @s_maabc
      AND a~mtart IN @s_mtart
      AND a~matkl IN @s_matkl
      AND a~lvorm =  @space.
  IF gt_marc IS INITIAL.
    MESSAGE 'Keine Materialien zur Selektion gefunden.' TYPE 'E'.
  ENDIF.

  " Bewertungskreis je Zeile ergaenzen (fuer MBEW-Zugriff)
  LOOP AT gt_marc ASSIGNING FIELD-SYMBOL(<ls_marc>).
    READ TABLE gt_plants ASSIGNING <ls_plant>
      WITH KEY werks = <ls_marc>-werks BINARY SEARCH.
    IF sy-subrc = 0.
      <ls_marc>-bwkey = <ls_plant>-bwkey.
    ENDIF.
  ENDLOOP.

* --- 2) Hard-Filter: Dispobereich (MDMA, je Material/Werk) ------------
  IF p_fmrp = 'X'.
    SELECT matnr, werks FROM mdma
      INTO TABLE @DATA(lt_mdma)
      FOR ALL ENTRIES IN @gt_marc
      WHERE matnr = @gt_marc-matnr
        AND werks = @gt_marc-werks.
    SORT lt_mdma BY matnr werks.
    DELETE ADJACENT DUPLICATES FROM lt_mdma COMPARING matnr werks.
    LOOP AT gt_marc ASSIGNING <ls_marc>.
      READ TABLE lt_mdma TRANSPORTING NO FIELDS
        WITH KEY matnr = <ls_marc>-matnr
                 werks = <ls_marc>-werks BINARY SEARCH.
      IF sy-subrc = 0.
        CLEAR <ls_marc>-matnr.          "markieren zum Loeschen
      ENDIF.
    ENDLOOP.
    DELETE gt_marc WHERE matnr IS INITIAL.
    IF gt_marc IS INITIAL.
      MESSAGE 'Nach Dispobereich-Filter keine Materialien uebrig.'
        TYPE 'E'.
    ENDIF.
  ENDIF.

* --- 3) Hard-Filter: alternative Mengeneinheiten (DB-Join) ------------
  IF p_fuom = 'X'.
    SELECT DISTINCT m~matnr
      FROM marm AS m
      INNER JOIN mara AS a ON a~matnr = m~matnr
      INTO TABLE @DATA(lt_altuom)
      FOR ALL ENTRIES IN @gt_marc
      WHERE m~matnr =  @gt_marc-matnr
        AND m~meinh <> a~meins.
    SORT lt_altuom.
    LOOP AT gt_marc ASSIGNING <ls_marc>.
      READ TABLE lt_altuom TRANSPORTING NO FIELDS
        WITH KEY table_line = <ls_marc>-matnr BINARY SEARCH.
      IF sy-subrc = 0.
        CLEAR <ls_marc>-matnr.
      ENDIF.
    ENDLOOP.
    DELETE gt_marc WHERE matnr IS INITIAL.
    IF gt_marc IS INITIAL.
      MESSAGE 'Nach Mengeneinheiten-Filter keine Materialien uebrig.'
        TYPE 'E'.
    ENDIF.
  ENDIF.

  SORT gt_marc BY matnr werks.

* --- 4) MBEW (je Material + Bewertungskreis) --------------------------
  DATA lt_mbew TYPE tt_mbew.
  SELECT matnr, bwkey, vprsv, stprs, verpr, peinh, lvorm
    FROM mbew
    INTO CORRESPONDING FIELDS OF TABLE @lt_mbew
    FOR ALL ENTRIES IN @gt_marc
    WHERE matnr = @gt_marc-matnr
      AND bwkey = @gt_marc-bwkey
      AND bwtar = @space.
  SORT lt_mbew BY matnr bwkey.
  DELETE ADJACENT DUPLICATES FROM lt_mbew COMPARING matnr bwkey.

* --- 5) Customizing-/Konfig-Tabellen ----------------------------------
  DATA: lt_t006  TYPE tt_t006,
        lt_t134  TYPE tt_t134,
        lt_t438a TYPE tt_t438a,
        lt_t399d TYPE tt_t399d.

  SELECT msehi, decan FROM t006
    INTO CORRESPONDING FIELDS OF TABLE @lt_t006.
  SELECT mtart FROM t134
    INTO CORRESPONDING FIELDS OF TABLE @lt_t134.
  SELECT dismm, disvf FROM t438a
    INTO CORRESPONDING FIELDS OF TABLE @lt_t438a.
  SELECT werks, bztek FROM t399d
    INTO CORRESPONDING FIELDS OF TABLE @lt_t399d
    WHERE werks IN @s_werks.
  SORT lt_t399d BY werks.

*----------------------------------------------------------------------*
* 6) Ausgabe je Reiter erzeugen
*----------------------------------------------------------------------*
* --- MATDOC: Bewegungsdaten paketiert und bei Excel-Grenze aufgeteilt --
* Bewusst inkl. Stornoreferenzen (SJAHR/SMBLN/SMBLP): die Storno-
* Bereinigung erfolgt in stoxion, daher hier roh exportieren.
  PERFORM reset_result_writer.

  DATA: lt_chunk         TYPE tt_key,
        lt_mov           TYPE tt_mov,
        lt_key_mov       TYPE tt_mov,
        lt_book_marc     TYPE tt_marc,
        lt_book_mov      TYPE tt_mov,
        lv_total         TYPE i,
        lv_from          TYPE i,
        lv_to            TYPE i,
        lv_mov_idx       TYPE i,
        lv_prev_idx      TYPE i,
        lv_key_rows      TYPE i,
        lv_book_mov_rows TYPE i,
        lv_max_data_rows TYPE i.

  lv_max_data_rows = gc_excel_max_rows - 2.
  lv_from = 1.

  WHILE lv_from <= lines( gt_marc ).
    CLEAR lt_chunk.
    lv_to = lv_from + gc_chunk - 1.
    LOOP AT gt_marc ASSIGNING <ls_marc>
         FROM lv_from TO lv_to.
      APPEND VALUE ty_marc_key( matnr = <ls_marc>-matnr
                                werks = <ls_marc>-werks ) TO lt_chunk.
    ENDLOOP.
    lv_from = lv_from + gc_chunk.
    IF lt_chunk IS INITIAL.
      EXIT.
    ENDIF.

    CLEAR lt_mov.
    SELECT m~matnr, m~werks, m~bwart, m~shkzg, m~menge,
           k~budat, m~mjahr, m~mblnr, m~zeile,
           m~sjahr, m~smbln, m~smblp
      FROM mseg AS m
      INNER JOIN mkpf AS k
        ON  k~mblnr = m~mblnr
        AND k~mjahr = m~mjahr
      INTO CORRESPONDING FIELDS OF TABLE @lt_mov
      FOR ALL ENTRIES IN @lt_chunk
      WHERE m~matnr =  @lt_chunk-matnr
        AND m~werks =  @lt_chunk-werks
        AND m~sobkz IN @s_sobkz
        AND m~bwart IN @s_bwart
        AND k~budat IN @s_budat.

    SORT lt_mov BY matnr werks budat mblnr zeile.

    LOOP AT lt_chunk ASSIGNING FIELD-SYMBOL(<ls_key>).
      CLEAR lt_key_mov.

      READ TABLE lt_mov TRANSPORTING NO FIELDS
        WITH KEY matnr = <ls_key>-matnr
                 werks = <ls_key>-werks
        BINARY SEARCH.
      IF sy-subrc = 0.
        lv_mov_idx = sy-tabix.
        WHILE lv_mov_idx > 1.
          lv_prev_idx = lv_mov_idx - 1.
          READ TABLE lt_mov ASSIGNING FIELD-SYMBOL(<ls_prev_mov>)
            INDEX lv_prev_idx.
          IF sy-subrc <> 0 OR
             <ls_prev_mov>-matnr <> <ls_key>-matnr OR
             <ls_prev_mov>-werks <> <ls_key>-werks.
            EXIT.
          ENDIF.
          lv_mov_idx = lv_prev_idx.
        ENDWHILE.

        LOOP AT lt_mov ASSIGNING FIELD-SYMBOL(<ls_mov>)
             FROM lv_mov_idx.
          IF <ls_mov>-matnr <> <ls_key>-matnr OR
             <ls_mov>-werks <> <ls_key>-werks.
            EXIT.
          ENDIF.
          APPEND <ls_mov> TO lt_key_mov.
        ENDLOOP.
      ENDIF.

      lv_key_rows = lines( lt_key_mov ).
      IF lv_key_rows > lv_max_data_rows.
        MESSAGE |Material/Werk { <ls_key>-matnr }/{ <ls_key>-werks } | &&
                |ueberschreitet alleine das Excel-Zeilenlimit | &&
                |({ lv_key_rows } Datenzeilen, max. { lv_max_data_rows }). | &&
                |Bitte Zeitraum weiter eingrenzen.| TYPE 'E'.
      ENDIF.

      IF lt_book_marc IS NOT INITIAL AND
         ( lv_book_mov_rows + lv_key_rows > lv_max_data_rows OR
           lines( lt_book_marc ) >= lv_max_data_rows ).
        PERFORM finalize_workbook USING lt_book_marc
                                        lt_book_mov
                                        gt_plants
                                        lt_t006
                                        lt_t134
                                        lt_t438a
                                        lt_t399d
                                        lt_mbew.
        CLEAR: lt_book_marc, lt_book_mov, lv_book_mov_rows.
      ENDIF.

      READ TABLE gt_marc ASSIGNING <ls_marc>
        WITH KEY matnr = <ls_key>-matnr
                 werks = <ls_key>-werks
        BINARY SEARCH.
      IF sy-subrc = 0.
        APPEND <ls_marc> TO lt_book_marc.
      ENDIF.

      APPEND LINES OF lt_key_mov TO lt_book_mov.
      lv_book_mov_rows = lv_book_mov_rows + lv_key_rows.
      lv_total = lv_total + lv_key_rows.
    ENDLOOP.
  ENDWHILE.

  IF lt_book_marc IS NOT INITIAL.
    PERFORM finalize_workbook USING lt_book_marc
                                    lt_book_mov
                                    gt_plants
                                    lt_t006
                                    lt_t134
                                    lt_t438a
                                    lt_t399d
                                    lt_mbew.
    CLEAR: lt_book_marc, lt_book_mov, lv_book_mov_rows.
  ENDIF.
  CLEAR: lt_chunk, lt_mov, lt_key_mov.

  PERFORM write_export_result.

          |{ lines( gt_marc ) } Material/Werk-Kombinationen, | &&
          |{ lv_total } Bewegungen, | &&
          |{ gv_workbook_count } Arbeitsmappe(n).| TYPE 'S'.

  CLEAR: gt_sheets,

*&---------------------------------------------------------------------*
*& FORM finalize_workbook - Paket als komplette Arbeitsmappe erzeugen
*&---------------------------------------------------------------------*
FORM finalize_workbook USING it_book_marc TYPE tt_marc
                             it_book_mov  TYPE tt_mov
                             it_plants    TYPE tt_plant
                             it_t006      TYPE tt_t006
                             it_t134      TYPE tt_t134
                             it_t438a     TYPE tt_t438a
                             it_t399d     TYPE tt_t399d
                             it_mbew_all  TYPE tt_mbew.

  DATA: lt_book_plants TYPE tt_plant,
        lt_book_mara   TYPE tt_mara,
        lt_book_mbew   TYPE tt_mbew,
        lv_xlsx        TYPE xstring.

  CLEAR gt_sheets.

  LOOP AT it_book_marc ASSIGNING FIELD-SYMBOL(<ls_book_marc>).
    READ TABLE it_plants ASSIGNING FIELD-SYMBOL(<ls_book_plant>)
      WITH KEY werks = <ls_book_marc>-werks.
    IF sy-subrc = 0.
      APPEND <ls_book_plant> TO lt_book_plants.
    ENDIF.

    APPEND VALUE ty_mara( matnr = <ls_book_marc>-matnr
                          mtart = <ls_book_marc>-mtart
                          lvorm = space
                          meins = <ls_book_marc>-meins ) TO lt_book_mara.
  ENDLOOP.

  SORT lt_book_plants BY werks.
  DELETE ADJACENT DUPLICATES FROM lt_book_plants COMPARING werks.
  SORT lt_book_mara BY matnr.
  DELETE ADJACENT DUPLICATES FROM lt_book_mara COMPARING matnr.

  LOOP AT it_book_marc ASSIGNING <ls_book_marc>.
    READ TABLE it_mbew_all ASSIGNING FIELD-SYMBOL(<ls_book_mbew>)
      WITH KEY matnr = <ls_book_marc>-matnr
               bwkey = <ls_book_marc>-bwkey
      BINARY SEARCH.
    IF sy-subrc = 0.
      APPEND <ls_book_mbew> TO lt_book_mbew.
    ENDIF.
  ENDLOOP.
  SORT lt_book_mbew BY matnr bwkey.
  DELETE ADJACENT DUPLICATES FROM lt_book_mbew COMPARING matnr bwkey.

  PERFORM build_common_sheets USING lt_book_plants
                                    it_t006
                                    it_t134
                                    it_t438a
                                    it_t399d.
  PERFORM build_master_sheets USING it_book_marc
                                    lt_book_mara
                                    lt_book_mbew.
  PERFORM build_matdoc_sheet USING it_book_mov.
  PERFORM reorder_sheets_like_template.
  PERFORM build_xlsx_zip CHANGING lv_xlsx.
  PERFORM publish_workbook USING lv_xlsx.

  CLEAR gt_sheets.

ENDFORM.

*&---------------------------------------------------------------------*
*& FORM publish_workbook - ein XLSX sammeln oder ZIP vorbereiten
*&---------------------------------------------------------------------*
FORM publish_workbook USING iv_xlsx TYPE xstring.

  DATA lv_entry_name TYPE string.

  gv_workbook_count = gv_workbook_count + 1.
  PERFORM workbook_entry_name USING gv_workbook_count
                              CHANGING lv_entry_name.

  IF gv_workbook_count = 1.
    gv_single_xlsx = iv_xlsx.
    gv_single_name = lv_entry_name.
  ELSE.
    IF go_result_zip IS INITIAL.
      CREATE OBJECT go_result_zip.
      go_result_zip->add( name = gv_single_name content = gv_single_xlsx ).
      CLEAR gv_single_xlsx.
    ENDIF.
    go_result_zip->add( name = lv_entry_name content = iv_xlsx ).
  ENDIF.

ENDFORM.

FORM workbook_entry_name USING iv_index TYPE i
                         CHANGING cv_name TYPE string.

  DATA lv_index TYPE string.

  IF iv_index < 10.
    lv_index = |00{ iv_index }|.
  ELSEIF iv_index < 100.
    lv_index = |0{ iv_index }|.
  ELSE.
    lv_index = iv_index.
    CONDENSE lv_index NO-GAPS.
  ENDIF.

  cv_name = |stoxion_export_{ lv_index }.xlsx|.

ENDFORM.

*&---------------------------------------------------------------------*
*& FORM write_export_result - einzelnes XLSX oder ZIP schreiben
*&---------------------------------------------------------------------*
FORM write_export_result.

  DATA: lv_result_file TYPE string,
        lv_zip         TYPE xstring.

  IF gv_workbook_count = 0.
    MESSAGE 'Keine Arbeitsmappe fuer XLSX-Erzeugung vorhanden.' TYPE 'E'.
  ENDIF.

  PERFORM build_workbook_filename CHANGING gv_xlsx_file.

  IF gv_workbook_count = 1.
    PERFORM write_binary_file USING gv_xlsx_file gv_single_xlsx.
  ELSE.
    IF go_result_zip IS INITIAL.
      MESSAGE 'ZIP-Erzeugung nicht initialisiert.' TYPE 'E'.
    ENDIF.
    lv_zip = go_result_zip->save( ).
    PERFORM build_zip_filename USING gv_xlsx_file
                               CHANGING lv_result_file.
    gv_xlsx_file = lv_result_file.
    PERFORM write_binary_file USING gv_xlsx_file lv_zip.
  ENDIF.

ENDFORM.

FORM build_zip_filename USING iv_xlsx_file TYPE string
                        CHANGING cv_zip_file TYPE string.

  DATA: lv_len      TYPE i,
        lv_base_len TYPE i.

  cv_zip_file = iv_xlsx_file.
  lv_len = strlen( cv_zip_file ).
  IF lv_len > 5.
    lv_base_len = lv_len - 5.
    cv_zip_file = cv_zip_file(lv_base_len) && '.zip'.
  ELSE.
    cv_zip_file = cv_zip_file && '.zip'.
  ENDIF.

ENDFORM.

FORM write_binary_file USING iv_file    TYPE string
                             iv_content TYPE xstring.

  DATA lt_bin TYPE solix_tab.

  IF p_serv = 'X'.
    CALL FUNCTION 'AUTHORITY_CHECK_DATASET'
      EXPORTING
        activity         = sabc_act_write
        filename         = iv_file
      EXCEPTIONS
        no_authority     = 1
        activity_unknown = 2
        OTHERS           = 3.
    IF sy-subrc <> 0.
      MESSAGE |Keine Schreibberechtigung fuer { iv_file }.| TYPE 'E'.
    ENDIF.

    OPEN DATASET iv_file FOR OUTPUT IN BINARY MODE.
    IF sy-subrc <> 0.
      MESSAGE |Datei { iv_file } nicht oeffenbar.| TYPE 'E'.
    ENDIF.
    TRANSFER iv_content TO iv_file.
    CLOSE DATASET iv_file.
  ELSE.
    lt_bin = cl_bcs_convert=>xstring_to_solix( iv_xstring = iv_content ).
    cl_gui_frontend_services=>gui_download(
      EXPORTING
        bin_filesize = xstrlen( iv_content )
        filename     = iv_file
        filetype     = 'BIN'
      CHANGING
        data_tab     = lt_bin
      EXCEPTIONS
        OTHERS       = 1 ).
    IF sy-subrc <> 0.
      MESSAGE |Download { iv_file } fehlgeschlagen (rc={ sy-subrc }).|
        TYPE 'E'.
    ENDIF.
  ENDIF.

ENDFORM.

*&---------------------------------------------------------------------*
*& Sheet-Writer: Reiter sammeln, am Ende als eine XLSX schreiben
*&---------------------------------------------------------------------*
FORM sheet_open USING iv_name TYPE string.

  gv_sh_name = iv_name.
  gv_sh_isopen = abap_true.
  CLEAR gt_sh_buffer.

ENDFORM.

FORM sheet_put USING it_lines TYPE STANDARD TABLE.

  APPEND LINES OF it_lines TO gt_sh_buffer.

ENDFORM.

FORM sheet_close.

  IF gv_sh_isopen = abap_false.
    RETURN.
  ENDIF.

  APPEND VALUE ty_sheet( name  = gv_sh_name
                         lines = gt_sh_buffer ) TO gt_sheets.
  CLEAR gt_sh_buffer.
  CLEAR gv_sh_name.
  gv_sh_isopen = abap_false.

ENDFORM.

*&---------------------------------------------------------------------*
*& FORM write_sheet - komplette Reiter-Datei in einem Zug
*&---------------------------------------------------------------------*
FORM write_sheet USING iv_name  TYPE string
                       it_lines TYPE STANDARD TABLE.

  PERFORM sheet_open  USING iv_name.
  PERFORM sheet_put   USING it_lines.
  PERFORM sheet_close.

ENDFORM.

*&---------------------------------------------------------------------*
*& FORM reorder_sheets_like_template - Reiterfolge wie MVP-Template
*&---------------------------------------------------------------------*
FORM reorder_sheets_like_template.

  DATA lt_ordered TYPE tt_sheet.

  PERFORM append_ordered_sheet USING 'T001'     CHANGING lt_ordered.
  PERFORM append_ordered_sheet USING 'T001W'    CHANGING lt_ordered.
  PERFORM append_ordered_sheet USING 'T006'     CHANGING lt_ordered.
  PERFORM append_ordered_sheet USING 'T134'     CHANGING lt_ordered.
  PERFORM append_ordered_sheet USING 'V438A'    CHANGING lt_ordered.
  PERFORM append_ordered_sheet USING 'V_399D_E' CHANGING lt_ordered.
  PERFORM append_ordered_sheet USING 'MARA'     CHANGING lt_ordered.
  PERFORM append_ordered_sheet USING 'MATDOC'   CHANGING lt_ordered.
  PERFORM append_ordered_sheet USING 'MARC'     CHANGING lt_ordered.
  PERFORM append_ordered_sheet USING 'MBEW'     CHANGING lt_ordered.

  LOOP AT gt_sheets ASSIGNING FIELD-SYMBOL(<ls_sheet>).
    READ TABLE lt_ordered TRANSPORTING NO FIELDS
      WITH KEY name = <ls_sheet>-name.
    IF sy-subrc <> 0.
      APPEND <ls_sheet> TO lt_ordered.
    ENDIF.
  ENDLOOP.

  gt_sheets = lt_ordered.

ENDFORM.

FORM append_ordered_sheet USING iv_name TYPE string
                       CHANGING ct_sheets TYPE tt_sheet.

  READ TABLE gt_sheets ASSIGNING FIELD-SYMBOL(<ls_sheet>)
    WITH KEY name = iv_name.
  IF sy-subrc = 0.
    APPEND <ls_sheet> TO ct_sheets.
  ENDIF.

ENDFORM.

*&---------------------------------------------------------------------*
*& FORM validate_target_path - Zielparameter vor dem Export pruefen
*&---------------------------------------------------------------------*
FORM validate_target_path.

  IF p_path IS INITIAL.
    MESSAGE 'Bitte Zielordner, XLSX- oder ZIP-Datei angeben.' TYPE 'E'.
  ENDIF.

  IF p_serv = 'X' AND ( p_path CS ':\' OR p_path CS ':/' ).
    MESSAGE 'Fuer Server-Export bitte AL11-/Unix-Pfad angeben, ' &&
            'z.B. /tmp/stoxion/stoxion_export.xlsx.' TYPE 'E'.
  ENDIF.

ENDFORM.

*&---------------------------------------------------------------------*
*& FORM build_workbook_filename - Zielpfad + XLSX-Dateiname
*&---------------------------------------------------------------------*
FORM build_workbook_filename CHANGING cv_file TYPE string.

  DATA(lv_sep) = COND string( WHEN p_front = 'X' THEN '\' ELSE '/' ).

  cv_file = p_path.
  IF cv_file IS INITIAL.
    MESSAGE 'Bitte Zielordner, XLSX- oder ZIP-Datei angeben.' TYPE 'E'.
  ENDIF.

  DATA(lv_lower) = to_lower( val = cv_file ).
  IF lv_lower CP '*.xlsx'.
    RETURN.
  ENDIF.
  IF lv_lower CP '*.zip'.
    DATA(lv_zip_base_len) = strlen( cv_file ) - 4.
    cv_file = cv_file(lv_zip_base_len) && '.xlsx'.
    RETURN.
  ENDIF.

  DATA(lv_len) = strlen( cv_file ) - 1.
  IF cv_file+lv_len(1) <> lv_sep.
    cv_file = cv_file && lv_sep.
  ENDIF.
  cv_file = |{ cv_file }stoxion_export.xlsx|.

ENDFORM.

*&---------------------------------------------------------------------*
*& FORM write_xlsx_workbook - Workbook erzeugen und binaer speichern
*&---------------------------------------------------------------------*
FORM write_xlsx_workbook.

  DATA lv_xlsx TYPE xstring.

  IF gt_sheets IS INITIAL.
    MESSAGE 'Keine Reiterdaten fuer XLSX-Erzeugung vorhanden.' TYPE 'E'.
  ENDIF.

  PERFORM build_workbook_filename CHANGING gv_xlsx_file.
  PERFORM build_xlsx_zip CHANGING lv_xlsx.
  PERFORM write_binary_file USING gv_xlsx_file lv_xlsx.

ENDFORM.

*&---------------------------------------------------------------------*
*& FORM build_xlsx_zip - minimale OpenXML-Arbeitsmappe bauen
*&---------------------------------------------------------------------*
FORM build_xlsx_zip CHANGING cv_xlsx TYPE xstring.

  DATA: lo_zip         TYPE REF TO cl_abap_zip,
        lv_content     TYPE string,
        lv_workbook    TYPE string,
        lv_rels        TYPE string,
        lv_root_rels   TYPE string,
        lv_sheet_xml   TYPE string,
        lv_sheet_name  TYPE string,
        lv_part_name   TYPE string,
        lv_sheet_no    TYPE i,
        lv_sheet_no_c  TYPE string.

  CREATE OBJECT lo_zip.

  lv_content =
    |<?xml version="1.0" encoding="UTF-8" standalone="yes"?>| &&
    |<Types xmlns="http://schemas.openxmlformats.org/package/2006/content-types">| &&
    |<Default Extension="rels" ContentType="application/vnd.openxmlformats-package.relationships+xml"/>| &&
    |<Default Extension="xml" ContentType="application/xml"/>| &&
    |<Override PartName="/xl/workbook.xml" ContentType="application/vnd.openxmlformats-officedocument.spreadsheetml.sheet.main+xml"/>|.

  lv_workbook =
    |<?xml version="1.0" encoding="UTF-8" standalone="yes"?>| &&
    |<workbook xmlns="http://schemas.openxmlformats.org/spreadsheetml/2006/main" | &&
    |xmlns:r="http://schemas.openxmlformats.org/officeDocument/2006/relationships"><sheets>|.

  lv_rels =
    |<?xml version="1.0" encoding="UTF-8" standalone="yes"?>| &&
    |<Relationships xmlns="http://schemas.openxmlformats.org/package/2006/relationships">|.

  lv_sheet_no = 0.
  LOOP AT gt_sheets ASSIGNING FIELD-SYMBOL(<ls_sheet>).
    lv_sheet_no = lv_sheet_no + 1.
    lv_sheet_no_c = lv_sheet_no.
    CONDENSE lv_sheet_no_c NO-GAPS.
    PERFORM xml_escape USING <ls_sheet>-name CHANGING lv_sheet_name.

    lv_content = lv_content &&
      |<Override PartName="/xl/worksheets/sheet{ lv_sheet_no_c }.xml" | &&
      |ContentType="application/vnd.openxmlformats-officedocument.spreadsheetml.worksheet+xml"/>|.

    lv_workbook = lv_workbook &&
      |<sheet name="{ lv_sheet_name }" sheetId="{ lv_sheet_no_c }" | &&
      |r:id="rId{ lv_sheet_no_c }"/>|.

    lv_rels = lv_rels &&
      |<Relationship Id="rId{ lv_sheet_no_c }" | &&
      |Type="http://schemas.openxmlformats.org/officeDocument/2006/relationships/worksheet" | &&
      |Target="worksheets/sheet{ lv_sheet_no_c }.xml"/>|.

    PERFORM build_worksheet_xml USING <ls_sheet>-lines CHANGING lv_sheet_xml.
    lv_part_name = |xl/worksheets/sheet{ lv_sheet_no_c }.xml|.
    PERFORM add_zip_string USING lo_zip lv_part_name lv_sheet_xml.
  ENDLOOP.

  lv_content = lv_content && |</Types>|.
  lv_workbook = lv_workbook && |</sheets></workbook>|.
  lv_rels = lv_rels && |</Relationships>|.

  PERFORM add_zip_string USING lo_zip '[Content_Types].xml' lv_content.
  PERFORM add_zip_string USING lo_zip 'xl/workbook.xml' lv_workbook.
  PERFORM add_zip_string USING lo_zip 'xl/_rels/workbook.xml.rels' lv_rels.
  lv_root_rels =
    |<?xml version="1.0" encoding="UTF-8" standalone="yes"?>| &&
    |<Relationships xmlns="http://schemas.openxmlformats.org/package/2006/relationships">| &&
    |<Relationship Id="rId1" Type="http://schemas.openxmlformats.org/officeDocument/2006/relationships/officeDocument" Target="xl/workbook.xml"/>| &&
    |</Relationships>|.
  PERFORM add_zip_string USING lo_zip '_rels/.rels' lv_root_rels.

  cv_xlsx = lo_zip->save( ).

ENDFORM.

FORM build_worksheet_xml USING it_lines TYPE tt_string
                      CHANGING cv_xml   TYPE string.

  DATA: lt_cells     TYPE STANDARD TABLE OF string,
        lv_row       TYPE i,
        lv_col       TYPE i,
        lv_col_name  TYPE string,
        lv_cell_ref  TYPE string,
        lv_cell_xml  TYPE string,
        lv_value     TYPE string,
        lv_row_c     TYPE string.

  cv_xml =
    |<?xml version="1.0" encoding="UTF-8" standalone="yes"?>| &&
    |<worksheet xmlns="http://schemas.openxmlformats.org/spreadsheetml/2006/main">| &&
    |<sheetData>|.

  lv_row = 0.
  LOOP AT it_lines ASSIGNING FIELD-SYMBOL(<lv_line>).
    lv_row = lv_row + 1.
    lv_row_c = lv_row.
    CONDENSE lv_row_c NO-GAPS.
    SPLIT <lv_line> AT gc_tab INTO TABLE lt_cells.
    cv_xml = cv_xml && |<row r="{ lv_row_c }">|.

    lv_col = 0.
    LOOP AT lt_cells ASSIGNING FIELD-SYMBOL(<lv_cell>).
      lv_col = lv_col + 1.
      PERFORM column_name USING lv_col CHANGING lv_col_name.
      lv_cell_ref = |{ lv_col_name }{ lv_row_c }|.
      PERFORM xml_escape USING <lv_cell> CHANGING lv_value.
      lv_cell_xml =
        |<c r="{ lv_cell_ref }" t="inlineStr"><is><t>{ lv_value }</t></is></c>|.
      cv_xml = cv_xml && lv_cell_xml.
    ENDLOOP.

    cv_xml = cv_xml && |</row>|.
  ENDLOOP.

  cv_xml = cv_xml && |</sheetData></worksheet>|.

ENDFORM.

FORM column_name USING iv_col TYPE i CHANGING cv_name TYPE string.

  DATA: lv_col TYPE i,
        lv_mod TYPE i,
        lv_chr TYPE c LENGTH 1.

  CLEAR cv_name.
  lv_col = iv_col.
  WHILE lv_col > 0.
    lv_mod = ( lv_col - 1 ) MOD 26.
    lv_chr = sy-abcde+lv_mod(1).
    cv_name = lv_chr && cv_name.
    lv_col = ( lv_col - 1 ) DIV 26.
  ENDWHILE.

ENDFORM.

FORM xml_escape USING iv_value TYPE string CHANGING cv_value TYPE string.

  cv_value = iv_value.
  REPLACE ALL OCCURRENCES OF '&' IN cv_value WITH '&amp;'.
  REPLACE ALL OCCURRENCES OF '<' IN cv_value WITH '&lt;'.
  REPLACE ALL OCCURRENCES OF '>' IN cv_value WITH '&gt;'.
  REPLACE ALL OCCURRENCES OF '"' IN cv_value WITH '&quot;'.
  REPLACE ALL OCCURRENCES OF '''' IN cv_value WITH '&apos;'.

ENDFORM.

FORM add_zip_string USING io_zip  TYPE REF TO cl_abap_zip
                          iv_name TYPE string
                          iv_text TYPE string.

  DATA: lo_conv    TYPE REF TO cl_abap_conv_out_ce,
        lv_content TYPE xstring.

  lo_conv = cl_abap_conv_out_ce=>create( encoding = 'UTF-8' ).
  lo_conv->write( data = iv_text ).
  lv_content = lo_conv->get_buffer( ).
  io_zip->add( name = iv_name content = lv_content ).

ENDFORM.

*&---------------------------------------------------------------------*
*& FORM schedule_job - aktuellen Selektionsstand als Job einplanen
*&---------------------------------------------------------------------*
FORM schedule_job.

  DATA: lt_par      TYPE TABLE OF rsparams,
        lv_jobname  TYPE tbtcjob-jobname,
        lv_jobcount TYPE tbtcjob-jobcount.

  " Aktuelle Selektionswerte einsammeln
  CALL FUNCTION 'RS_REFRESH_FROM_SELECTOPTIONS'
    EXPORTING
      curr_report     = sy-repid
    TABLES
      selection_table = lt_par
    EXCEPTIONS
      OTHERS          = 1.
  IF sy-subrc <> 0.
    MESSAGE 'Selektionswerte konnten nicht gelesen werden.' TYPE 'E'.
  ENDIF.

  " Im Job zwingend Server-Export
  LOOP AT lt_par ASSIGNING FIELD-SYMBOL(<ls_par>).
    CASE <ls_par>-selname.
      WHEN 'P_FRONT'. <ls_par>-low = space.
      WHEN 'P_SERV'.  <ls_par>-low = 'X'.
    ENDCASE.
  ENDLOOP.

  " Plausibilitaet: Pfad sollte ein Server-Pfad sein
  IF p_path CS ':\' OR p_path CS ':/'.
    MESSAGE 'Hinweis: Fuer den Job einen Server-Pfad (AL11) angeben, ' &&
            'z.B. /tmp/stoxion/ - aktueller Pfad sieht nach lokalem ' &&
            'Laufwerk aus.' TYPE 'E'.
  ENDIF.

  lv_jobname = |ZSTOXION_{ sy-datum }_{ sy-uzeit }|.

  CALL FUNCTION 'JOB_OPEN'
    EXPORTING
      jobname          = lv_jobname
    IMPORTING
      jobcount         = lv_jobcount
    EXCEPTIONS
      cant_create_job  = 1
      invalid_job_data = 2
      jobname_missing  = 3
      OTHERS           = 4.
  IF sy-subrc <> 0.
    MESSAGE 'Job konnte nicht angelegt werden.' TYPE 'E'.
  ENDIF.

  SUBMIT (sy-repid)
    WITH SELECTION-TABLE lt_par
    VIA JOB lv_jobname NUMBER lv_jobcount
    AND RETURN.

  CALL FUNCTION 'JOB_CLOSE'
    EXPORTING
      jobname              = lv_jobname
      jobcount             = lv_jobcount
      strtimmed            = 'X'                "Sofortstart
    EXCEPTIONS
      cant_start_immediate = 1
      invalid_startdate    = 2
      jobname_missing      = 3
      job_close_failed     = 4
      job_nosteps          = 5
      job_notex            = 6
      lock_failed          = 7
      OTHERS               = 8.
  IF sy-subrc <> 0.
    MESSAGE 'Job konnte nicht freigegeben werden (SM37 pruefen).'
      TYPE 'E'.
  ENDIF.

  MESSAGE |Job { lv_jobname } eingeplant (Sofortstart). | &&
          |Ueberwachung in SM37, Dateien danach in AL11.| TYPE 'S'.

ENDFORM.
