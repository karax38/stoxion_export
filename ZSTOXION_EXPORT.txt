*&---------------------------------------------------------------------*
*& Report  ZSTOXION_EXPORT
*&---------------------------------------------------------------------*
*& Zweck   : Template-konformer SAP-Datenexport fuer die stoxion
*&           Safety-Stock-Optimierung.
*&
*& Ausgabe  : Eine oder mehrere XLSX-Arbeitsmappen mit einem Reiter je
*&           fachlichem Datenbereich. Jeder Reiter enthaelt zwei Kopfzeilen:
*&           Zeile 1 = fachliche Spaltenbezeichnung
*&           Zeile 2 = technischer Feldname
*&           ab Zeile 3 = Daten
*&
*& Reiter   : T001, T001W, T006, T134, V438A, V_399D_E,
*&            MARA, MATDOC, MARC, MBEW, MDMA, MDLV
*&
*& Quellen  : Material- und Werksstammdaten aus MARA/MARC/MBEW,
*&           Bewegungen aus MSEG/MKPF, Dispobereiche aus MDMA/MDLV
*&           sowie die benoetigten Organisations- und Customizingtabellen.
*&
*& Dispobereiche:
*&   - MARC-DIBER kennzeichnet, ob zu MATNR/WERKS MDMA-Eintraege
*&     vorhanden sind.
*&   - Ohne Dispobereich kommen DISMM/EISBE/PLIFZ/PSTAT aus MARC.
*&   - Mit Dispobereich kommen DISMM/EISBE/PLIFZ/PSTAT/LOEKZ aus MDMA.
*&   - MDLV liefert die gueltigen BERID/WERKS/LGORT-Zuordnungen.
*&
*& Format   : Keine Format-/Headerformatierung im Workbook. Alle Zellen
*&           werden als Text geschrieben, damit IDs und fuehrende Nullen
*&           unveraendert bleiben. Alle Reiter frieren die ersten zwei
*&           Zeilen ein und setzen den AutoFilter von Zeile 2 bis zur
*&           letzten ausgegebenen Spalte und Datenzeile.
*&
*& Betrieb  : Fuer groessere Exporte den Hintergrundjob verwenden und die
*&           Datei danach vom Applikationsserver abholen. Es werden keine
*&           personenbezogenen Daten exportiert.
*&---------------------------------------------------------------------*
REPORT zstoxion_export.

           gc_excel_max_rows TYPE i VALUE 1048576,  "Excel-Zeilenlimit (Obergrenze P_MAXROW)
           demand_history_months_max      TYPE i VALUE 36,
           demand_history_months_fallback TYPE i VALUE 24.

                  s_sobkz FOR mseg-sobkz.              "Sonderbestand (opt.)

  PARAMETERS: p_path  TYPE string LOWER CASE
                      DEFAULT 'C:\stoxion\stoxion_export.xlsx'. "Ziel
  PARAMETERS: p_maxrow TYPE i DEFAULT 500000.  "Max. Zeilen je Arbeitsmappe
SELECTION-SCREEN END OF BLOCK b4.

*----------------------------------------------------------------------*
INITIALIZATION.
*----------------------------------------------------------------------*
  CASE sy-langu.
    WHEN 'D'.
      sscrfields-functxt_01 = 'Job einplanen'.

      gv_t_org = 'Organisation und Material'.
      gv_t_mov = 'Bewegungsdaten'.
      gv_t_flt = 'Filter und Dispobereiche'.
      gv_t_tgt = 'Exportziel'.

      %_s_werks_%_app_%-text = 'Werk(e)'.
      %_s_matnr_%_app_%-text = 'Material'.
      %_s_mtart_%_app_%-text = 'Materialart'.
      %_s_matkl_%_app_%-text = 'Warengruppe'.
      %_s_dismm_%_app_%-text = 'Dispomerkmal'.
      %_s_maabc_%_app_%-text = 'ABC-Kennzeichen'.
      %_s_budat_%_app_%-text = 'Buchungsdatum'.
      %_s_bwart_%_app_%-text = 'Bewegungsart'.
      %_s_sobkz_%_app_%-text = 'Sonderbestand'.
      %_p_fmrp_%_app_%-text = 'Dispobereiche exportieren'.
      %_p_fuom_%_app_%-text = 'Alternative ME ausschliessen'.
      %_p_front_%_app_%-text = 'Frontend-Download'.
      %_p_serv_%_app_%-text = 'Applikationsserver'.
      %_p_path_%_app_%-text = 'Zielordner, XLSX- oder ZIP-Datei'.
      %_p_maxrow_%_app_%-text = 'Max. Zeilen je Arbeitsmappe'.

    WHEN OTHERS.
      sscrfields-functxt_01 = 'Schedule job'.

      gv_t_org = 'Organization and material'.
      gv_t_mov = 'Movement data'.
      gv_t_flt = 'Filters and MRP areas'.
      gv_t_tgt = 'Export target'.

      %_s_werks_%_app_%-text = 'Plant(s)'.
      %_s_matnr_%_app_%-text = 'Material'.
      %_s_mtart_%_app_%-text = 'Material type'.
      %_s_matkl_%_app_%-text = 'Material group'.
      %_s_dismm_%_app_%-text = 'MRP type'.
      %_s_maabc_%_app_%-text = 'ABC indicator'.
      %_s_budat_%_app_%-text = 'Posting date'.
      %_s_bwart_%_app_%-text = 'Movement type'.
      %_s_sobkz_%_app_%-text = 'Special stock'.
      %_p_fmrp_%_app_%-text = 'Export MRP areas'.
      %_p_fuom_%_app_%-text = 'Exclude alternative UoM'.
      %_p_front_%_app_%-text = 'Frontend download'.
      %_p_serv_%_app_%-text = 'Application server'.
      %_p_path_%_app_%-text = 'Target folder, XLSX or ZIP file'.
      %_p_maxrow_%_app_%-text = 'Max rows per workbook'.
  ENDCASE.

  " Defaults nur im Dialog setzen: Bei SUBMIT ... VIA JOB laeuft
  " INITIALIZATION vor der Uebernahme der Selektionstabelle, und
  " RS_REFRESH_FROM_SELECTOPTIONS liefert fuer leere Select-Options
  " keine Zeilen. Eine im Dialog bewusst geleerte Selektion (z.B.
  " "alle Bewegungsarten") wuerde im Job sonst still re-defaulted.
  IF sy-batch IS INITIAL.
    DATA: lv_init_window_end      TYPE sy-datum,
          lv_init_start_max       TYPE sy-datum,
          lv_init_start_fallback  TYPE sy-datum.

    PERFORM calc_demand_history_window
      CHANGING lv_init_window_end
               lv_init_start_max
               lv_init_start_fallback.

    PERFORM set_demand_history_budat
      USING lv_init_start_max
            lv_init_window_end.

    " Default fuer typische Nachfrage- und Umlagerungsbewegungen.
    s_bwart-sign   = 'I'.
    s_bwart-option = 'EQ'.
    s_bwart-low    = '201'. APPEND s_bwart.
    s_bwart-low    = '261'. APPEND s_bwart.
    s_bwart-low    = '281'. APPEND s_bwart.
    s_bwart-low    = '543'. APPEND s_bwart.
    s_bwart-low    = '551'. APPEND s_bwart.
    s_bwart-low    = '601'. APPEND s_bwart.
    s_bwart-low    = '633'. APPEND s_bwart.
    s_bwart-low    = '641'. APPEND s_bwart.
    s_bwart-low    = '643'. APPEND s_bwart.
    s_bwart-low    = '645'. APPEND s_bwart.
  ENDIF.

*----------------------------------------------------------------------*
AT SELECTION-SCREEN.
*----------------------------------------------------------------------*
  IF sscrfields-ucomm = 'FC01'.
    PERFORM schedule_job.
  ENDIF.

*----------------------------------------------------------------------*
START-OF-SELECTION.
*----------------------------------------------------------------------*

* --- Batch-Schutz: Frontend-Download geht im Hintergrund nicht --------
  IF sy-batch = 'X' AND p_front = 'X'.
    p_front = space.
    p_serv  = 'X'.
    MESSAGE 'Hintergrundlauf: Ausgabe auf Server-Export umgestellt.'
      TYPE 'S'.
  ENDIF.

  PERFORM validate_target_path.
  PERFORM validate_max_rows.

* --- 0) Werke aufloesen, Berechtigungen, Org-Daten --------------------
  SELECT werks, bwkey FROM t001w
    INTO CORRESPONDING FIELDS OF TABLE @gt_plants
    WHERE werks IN @s_werks.
  IF gt_plants IS INITIAL.
    MESSAGE 'Kein Werk zur Selektion gefunden.' TYPE 'E'.
  ENDIF.

  LOOP AT gt_plants ASSIGNING FIELD-SYMBOL(<ls_plant>).
    AUTHORITY-CHECK OBJECT 'M_MATE_WRK'
      ID 'ACTVT' FIELD '03'
      ID 'WERKS' FIELD <ls_plant>-werks.
    IF sy-subrc <> 0.
      MESSAGE |Keine Berechtigung fuer Werk { <ls_plant>-werks }.|
        TYPE 'E'.
    ENDIF.

    SELECT SINGLE bukrs FROM t001k
      INTO @<ls_plant>-bukrs
      WHERE bwkey = @<ls_plant>-bwkey.
    IF sy-subrc <> 0.
      MESSAGE |Kein Buchungskreis zu Werk { <ls_plant>-werks }.| TYPE 'E'.
    ENDIF.
    SELECT SINGLE waers FROM t001
      INTO @<ls_plant>-waers
      WHERE bukrs = @<ls_plant>-bukrs.
    IF sy-subrc <> 0.
      MESSAGE |Buchungskreis { <ls_plant>-bukrs } nicht gefunden.|
        TYPE 'E'.
    ENDIF.
  ENDLOOP.
  SORT gt_plants BY werks.

* --- 1) Kandidaten: MARC + MARA mit Stammdatenfiltern -----------------
  SELECT c~matnr, c~werks, c~lvorm, c~dismm, c~eisbe, c~plifz,
         c~webaz, c~pstat, c~maabc, c~wzeit,
         a~mtart, a~meins
    FROM marc AS c
    INNER JOIN mara AS a ON a~matnr = c~matnr
    INTO CORRESPONDING FIELDS OF TABLE @gt_marc
    WHERE c~werks IN @s_werks
      AND c~matnr IN @s_matnr
      AND c~lvorm =  @space
      AND c~maabc IN @s_maabc
      AND a~mtart IN @s_mtart
      AND a~matkl IN @s_matkl
      AND a~lvorm =  @space.
  IF gt_marc IS INITIAL.
    MESSAGE 'Keine Materialien zur Selektion gefunden.' TYPE 'E'.
  ENDIF.

  " Bewertungskreis je Zeile ergaenzen (fuer MBEW-Zugriff)
  LOOP AT gt_marc ASSIGNING FIELD-SYMBOL(<ls_marc>).
    READ TABLE gt_plants ASSIGNING <ls_plant>
      WITH KEY werks = <ls_marc>-werks BINARY SEARCH.
    IF sy-subrc = 0.
      <ls_marc>-bwkey = <ls_plant>-bwkey.
    ENDIF.
  ENDLOOP.

* --- 2) Dispobereiche (MDMA, je Material/Werk) kennzeichnen -----------
  DATA lt_mdma TYPE tt_mdma.
  SELECT matnr, berid, werks, dismm, eisbe, plifz, pstat, loekz
    FROM mdma
    INTO CORRESPONDING FIELDS OF TABLE @lt_mdma
    FOR ALL ENTRIES IN @gt_marc
    WHERE matnr = @gt_marc-matnr
      AND werks = @gt_marc-werks.
  SORT lt_mdma BY matnr werks berid.

  LOOP AT gt_marc ASSIGNING <ls_marc>.
    READ TABLE lt_mdma TRANSPORTING NO FIELDS
      WITH KEY matnr = <ls_marc>-matnr
               werks = <ls_marc>-werks BINARY SEARCH.
    IF sy-subrc = 0.
      <ls_marc>-diber = 'X'.
    ENDIF.
  ENDLOOP.

  IF s_dismm[] IS NOT INITIAL.
    LOOP AT gt_marc ASSIGNING <ls_marc>.
      DATA(lv_keep_mrp) = abap_false.

      IF <ls_marc>-diber = 'X'.
        LOOP AT lt_mdma ASSIGNING FIELD-SYMBOL(<ls_mdma_filter>)
             WHERE matnr = <ls_marc>-matnr
               AND werks = <ls_marc>-werks.
          IF <ls_mdma_filter>-dismm IN s_dismm.
            lv_keep_mrp = abap_true.
            EXIT.
          ENDIF.
        ENDLOOP.
      ELSEIF <ls_marc>-dismm IN s_dismm.
        lv_keep_mrp = abap_true.
      ENDIF.

      IF lv_keep_mrp = abap_false.
        CLEAR <ls_marc>-matnr.          "markieren zum Loeschen
      ENDIF.
    ENDLOOP.
    DELETE gt_marc WHERE matnr IS INITIAL.
    IF gt_marc IS INITIAL.
      MESSAGE 'Nach Dispomerkmal-Filter keine Materialien uebrig.'
        TYPE 'E'.
    ENDIF.
  ENDIF.

* --- 3) Materialien mit alternativen Mengeneinheiten ausschliessen ----
  IF p_fuom = 'X'.
    DATA lt_altuom TYPE SORTED TABLE OF mara-matnr
                         WITH UNIQUE KEY table_line.

    SELECT DISTINCT m~matnr
      FROM marm AS m
      INNER JOIN mara AS a ON a~matnr = m~matnr
      INTO TABLE @lt_altuom
      FOR ALL ENTRIES IN @gt_marc
      WHERE m~matnr =  @gt_marc-matnr
        AND m~meinh <> a~meins.

    LOOP AT gt_marc ASSIGNING <ls_marc>.
      READ TABLE lt_altuom TRANSPORTING NO FIELDS
        WITH TABLE KEY table_line = <ls_marc>-matnr.
      IF sy-subrc = 0.
        CLEAR <ls_marc>-matnr.
      ENDIF.
    ENDLOOP.
    DELETE gt_marc WHERE matnr IS INITIAL.
    IF gt_marc IS INITIAL.
      MESSAGE 'Nach Mengeneinheiten-Filter keine Materialien uebrig.'
        TYPE 'E'.
    ENDIF.
  ENDIF.

  SORT gt_marc BY matnr werks.

* --- 4) MBEW (je Material + Bewertungskreis) --------------------------
  DATA lt_mbew TYPE tt_mbew.
  SELECT matnr, bwkey, vprsv, stprs, verpr, peinh, lvorm
    FROM mbew
    INTO CORRESPONDING FIELDS OF TABLE @lt_mbew
    FOR ALL ENTRIES IN @gt_marc
    WHERE matnr = @gt_marc-matnr
      AND bwkey = @gt_marc-bwkey
      AND bwtar = @space.
  SORT lt_mbew BY matnr bwkey.
  DELETE ADJACENT DUPLICATES FROM lt_mbew COMPARING matnr bwkey.

* --- 5) Customizing-/Konfig-Tabellen ----------------------------------
  DATA: lt_t006  TYPE tt_t006,
        lt_t134  TYPE tt_t134,
        lt_t438a TYPE tt_t438a,
        lt_t399d TYPE tt_t399d,
        lt_mdlv  TYPE tt_mdlv.

  SELECT msehi, decan FROM t006
    INTO CORRESPONDING FIELDS OF TABLE @lt_t006.
  SELECT mtart FROM t134
    INTO CORRESPONDING FIELDS OF TABLE @lt_t134.
  SELECT dismm, disvf FROM t438a
    INTO CORRESPONDING FIELDS OF TABLE @lt_t438a.
  SELECT werks, bztek FROM t399d
    INTO CORRESPONDING FIELDS OF TABLE @lt_t399d
    WHERE werks IN @s_werks.
  SORT lt_t399d BY werks.
  PERFORM load_mdlv CHANGING lt_mdlv.

*----------------------------------------------------------------------*
* 6) Arbeitsmappen erzeugen
*----------------------------------------------------------------------*
* --- MATDOC: Bewegungsdaten paketiert, Aufteilung am Zeilenlimit -------
* Bewusst inkl. Stornoreferenzen (SJAHR/SMBLN/SMBLP): die Storno-
* Bereinigung erfolgt in stoxion, daher hier roh exportieren.
  PERFORM reset_result_writer.

  DATA: lt_chunk         TYPE tt_key,
        lt_mov           TYPE tt_mov,
        lt_key_mov       TYPE tt_mov,
        lt_book_marc     TYPE tt_marc,
        lt_book_mov      TYPE tt_mov,
        lv_history_window_end      TYPE sy-datum,
        lv_history_start_max       TYPE sy-datum,
        lv_history_start_fallback  TYPE sy-datum,
        lv_total         TYPE i,
        lv_from          TYPE i,
        lv_to            TYPE i,
        lv_mov_idx       TYPE i,
        lv_prev_idx      TYPE i,
        lv_key_rows      TYPE i,
        lv_book_mov_rows TYPE i,
        lv_max_data_rows TYPE i.

  PERFORM calc_demand_history_window
    CHANGING lv_history_window_end
             lv_history_start_max
             lv_history_start_fallback.
  PERFORM set_demand_history_budat
    USING lv_history_start_max
          lv_history_window_end.

  lv_max_data_rows = p_maxrow - 2.               "2 Kopfzeilen je Reiter
  lv_from = 1.

  WHILE lv_from <= lines( gt_marc ).
    CLEAR lt_chunk.
    lv_to = lv_from + gc_chunk - 1.
    LOOP AT gt_marc ASSIGNING <ls_marc>
         FROM lv_from TO lv_to.
      APPEND VALUE ty_marc_key( matnr = <ls_marc>-matnr
                                werks = <ls_marc>-werks ) TO lt_chunk.
    ENDLOOP.
    lv_from = lv_from + gc_chunk.
    IF lt_chunk IS INITIAL.
      EXIT.
    ENDIF.

    CLEAR lt_mov.
    SELECT m~matnr, m~werks, m~lgort, m~bwart, m~shkzg, m~menge,
           k~budat, m~mjahr, m~mblnr, m~zeile,
           m~sjahr, m~smbln, m~smblp
      FROM mseg AS m
      INNER JOIN mkpf AS k
        ON  k~mblnr = m~mblnr
        AND k~mjahr = m~mjahr
      INTO CORRESPONDING FIELDS OF TABLE @lt_mov
      FOR ALL ENTRIES IN @lt_chunk
      WHERE m~matnr =  @lt_chunk-matnr
        AND m~werks =  @lt_chunk-werks
        AND m~sobkz IN @s_sobkz
        AND m~bwart IN @s_bwart
        AND k~budat >= @lv_history_start_max
        AND k~budat <  @lv_history_window_end.

    SORT lt_mov BY matnr werks budat mblnr zeile.

    PERFORM filter_matdoc_history_window
      USING lv_history_window_end
            lv_history_start_max
            lv_history_start_fallback
            gt_marc
            lt_mdma
            lt_mdlv
      CHANGING lt_mov.

    LOOP AT lt_chunk ASSIGNING FIELD-SYMBOL(<ls_key>).
      CLEAR lt_key_mov.

      READ TABLE lt_mov TRANSPORTING NO FIELDS
        WITH KEY matnr = <ls_key>-matnr
                 werks = <ls_key>-werks
        BINARY SEARCH.
      IF sy-subrc = 0.
        lv_mov_idx = sy-tabix.
        WHILE lv_mov_idx > 1.
          lv_prev_idx = lv_mov_idx - 1.
          READ TABLE lt_mov ASSIGNING FIELD-SYMBOL(<ls_prev_mov>)
            INDEX lv_prev_idx.
          IF sy-subrc <> 0 OR
             <ls_prev_mov>-matnr <> <ls_key>-matnr OR
             <ls_prev_mov>-werks <> <ls_key>-werks.
            EXIT.
          ENDIF.
          lv_mov_idx = lv_prev_idx.
        ENDWHILE.

        LOOP AT lt_mov ASSIGNING FIELD-SYMBOL(<ls_mov>)
             FROM lv_mov_idx.
          IF <ls_mov>-matnr <> <ls_key>-matnr OR
             <ls_mov>-werks <> <ls_key>-werks.
            EXIT.
          ENDIF.
          APPEND <ls_mov> TO lt_key_mov.
        ENDLOOP.
      ENDIF.

      lv_key_rows = lines( lt_key_mov ).
      IF lv_key_rows > lv_max_data_rows.
        MESSAGE |Material/Werk { <ls_key>-matnr }/{ <ls_key>-werks } | &&
                |ueberschreitet alleine das Zeilenlimit je Arbeitsmappe | &&
                |({ lv_key_rows } Datenzeilen, max. { lv_max_data_rows }). | &&
                |Bitte Zeitraum eingrenzen oder P_MAXROW erhoehen.| TYPE 'E'.
      ENDIF.

      IF lt_book_marc IS NOT INITIAL AND
         ( lv_book_mov_rows + lv_key_rows > lv_max_data_rows OR
           lines( lt_book_marc ) >= lv_max_data_rows ).
        PERFORM finalize_workbook USING lt_book_marc
                                        lt_book_mov
                                        gt_plants
                                        lt_t006
                                        lt_t134
                                        lt_t438a
                                        lt_t399d
                                        lt_mdma
                                        lt_mdlv
                                        lt_mbew.
        CLEAR: lt_book_marc, lt_book_mov, lv_book_mov_rows.
      ENDIF.

      READ TABLE gt_marc ASSIGNING <ls_marc>
        WITH KEY matnr = <ls_key>-matnr
                 werks = <ls_key>-werks
        BINARY SEARCH.
      IF sy-subrc = 0.
        APPEND <ls_marc> TO lt_book_marc.
      ENDIF.

      APPEND LINES OF lt_key_mov TO lt_book_mov.
      lv_book_mov_rows = lv_book_mov_rows + lv_key_rows.
      lv_total = lv_total + lv_key_rows.
    ENDLOOP.
  ENDWHILE.

  IF lt_book_marc IS NOT INITIAL.
    PERFORM finalize_workbook USING lt_book_marc
                                    lt_book_mov
                                    gt_plants
                                    lt_t006
                                    lt_t134
                                    lt_t438a
                                    lt_t399d
                                    lt_mdma
                                    lt_mdlv
                                    lt_mbew.
    CLEAR: lt_book_marc, lt_book_mov, lv_book_mov_rows.
  ENDIF.
  CLEAR: lt_chunk, lt_mov, lt_key_mov.

  PERFORM write_export_result.

  MESSAGE |XLSX-Export abgeschlossen: { gv_xlsx_file }. | &&
          |{ lines( gt_marc ) } Material/Werk-Kombinationen, | &&
          |{ lv_total } Bewegungen, | &&
          |{ gv_workbook_count } Arbeitsmappe(n).| TYPE 'S'.

*&---------------------------------------------------------------------*
*& FORM calc_demand_history_window - Nachfragefenster berechnen
*&---------------------------------------------------------------------*
FORM calc_demand_history_window
  CHANGING cv_window_end      TYPE sy-datum
           cv_start_max       TYPE sy-datum
           cv_start_fallback  TYPE sy-datum.

  cv_window_end = sy-datum.
  cv_window_end+6(2) = '01'. "laufender Monat bleibt ausgeschlossen

  PERFORM subtract_months_from_first_day
    USING cv_window_end demand_history_months_max
    CHANGING cv_start_max.

  PERFORM subtract_months_from_first_day
    USING cv_window_end demand_history_months_fallback
    CHANGING cv_start_fallback.

ENDFORM.

*&---------------------------------------------------------------------*
*& FORM set_demand_history_budat - BUDAT-Selektion technisch setzen
*&---------------------------------------------------------------------*
FORM set_demand_history_budat USING iv_start_max  TYPE sy-datum
                                    iv_window_end TYPE sy-datum.

  CLEAR s_budat[].
  s_budat-sign   = 'I'.
  s_budat-option = 'BT'.
  s_budat-low    = iv_start_max.
  s_budat-high   = iv_window_end - 1.
  APPEND s_budat.

ENDFORM.

*&---------------------------------------------------------------------*
*& FORM subtract_months_from_first_day - Monatsersten verschieben
*&---------------------------------------------------------------------*
FORM subtract_months_from_first_day USING iv_date   TYPE sy-datum
                                          iv_months TYPE i
                                    CHANGING cv_date TYPE sy-datum.

  DATA: lv_year       TYPE i,
        lv_month      TYPE i,
        lv_year_txt   TYPE c LENGTH 4,
        lv_month_txt  TYPE n LENGTH 2.

  lv_year  = iv_date(4).
  lv_month = iv_date+4(2).

  DO iv_months TIMES.
    lv_month = lv_month - 1.
    IF lv_month = 0.
      lv_month = 12.
      lv_year = lv_year - 1.
    ENDIF.
  ENDDO.

  lv_year_txt = lv_year.
  lv_month_txt = lv_month.
  CONCATENATE lv_year_txt lv_month_txt '01' INTO cv_date.

ENDFORM.

*&---------------------------------------------------------------------*
*& FORM filter_matdoc_history_window - effektives MATDOC-Fenster
*&---------------------------------------------------------------------*
*& MATDOC wird global fuer maximal 36 volle Monate geladen. Das dritte
*& Jahr bleibt je MATNR/WERKS/BERID nur dann im Rohdatenexport, wenn nach
*& der Dispobereichszuordnung dort mindestens eine relevante Buchung
*& vorhanden ist. Es werden keine Kalenderwochen oder Nullwochen erzeugt.
*&---------------------------------------------------------------------*
FORM filter_matdoc_history_window USING iv_window_end      TYPE sy-datum
                                        iv_start_max       TYPE sy-datum
                                        iv_start_fallback  TYPE sy-datum
                                        it_marc            TYPE tt_marc
                                        it_mdma            TYPE tt_mdma
                                        it_mdlv            TYPE tt_mdlv
                                  CHANGING ct_mov          TYPE tt_mov.

  DATA: lt_third_year_keys TYPE tt_demand_key,
        lt_mdlv_by_lgort   TYPE tt_mdlv_by_lgort,
        ls_key             TYPE ty_demand_key,
        lv_berid           TYPE ty_berid,
        lv_index           TYPE sy-tabix.

  lt_mdlv_by_lgort = it_mdlv.

  DELETE ct_mov WHERE budat < iv_start_max
                   OR budat >= iv_window_end.

  LOOP AT ct_mov ASSIGNING FIELD-SYMBOL(<ls_mov_third_year>)
       WHERE budat < iv_start_fallback.
    CLEAR lv_berid.
    PERFORM determine_mov_berid
      USING <ls_mov_third_year>
            it_marc
            it_mdma
            lt_mdlv_by_lgort
      CHANGING lv_berid.

    ls_key-matnr = <ls_mov_third_year>-matnr.
    ls_key-werks = <ls_mov_third_year>-werks.
    ls_key-berid = lv_berid.
    INSERT ls_key INTO TABLE lt_third_year_keys.
  ENDLOOP.

  LOOP AT ct_mov ASSIGNING FIELD-SYMBOL(<ls_mov_history>).
    lv_index = sy-tabix.
    IF <ls_mov_history>-budat >= iv_start_fallback.
      CONTINUE.
    ENDIF.

    CLEAR lv_berid.
    PERFORM determine_mov_berid
      USING <ls_mov_history>
            it_marc
            it_mdma
            lt_mdlv_by_lgort
      CHANGING lv_berid.

    READ TABLE lt_third_year_keys TRANSPORTING NO FIELDS
      WITH TABLE KEY matnr = <ls_mov_history>-matnr
                     werks = <ls_mov_history>-werks
                     berid = lv_berid.
    IF sy-subrc <> 0.
      DELETE ct_mov INDEX lv_index.
    ENDIF.
  ENDLOOP.

ENDFORM.

*&---------------------------------------------------------------------*
*& FORM determine_mov_berid - MATDOC-Buchung einem Dispobereich zuordnen
*&---------------------------------------------------------------------*
FORM determine_mov_berid USING is_mov            TYPE ty_mov
                               it_marc           TYPE tt_marc
                               it_mdma           TYPE tt_mdma
                               it_mdlv_by_lgort  TYPE tt_mdlv_by_lgort
                         CHANGING cv_berid       TYPE ty_berid.

  DATA lv_candidate TYPE ty_berid.

  CLEAR cv_berid.

  READ TABLE it_marc ASSIGNING FIELD-SYMBOL(<ls_marc>)
    WITH KEY matnr = is_mov-matnr
             werks = is_mov-werks
    BINARY SEARCH.
  IF sy-subrc <> 0 OR <ls_marc>-diber <> 'X'.
    RETURN.
  ENDIF.

  IF is_mov-lgort IS NOT INITIAL.
    READ TABLE it_mdlv_by_lgort ASSIGNING FIELD-SYMBOL(<ls_mdlv_lgort>)
      WITH TABLE KEY werks = is_mov-werks
                     lgort = is_mov-lgort.
    IF sy-subrc = 0.
      lv_candidate = <ls_mdlv_lgort>-berid.
      READ TABLE it_mdma TRANSPORTING NO FIELDS
        WITH KEY matnr = is_mov-matnr
                 werks = is_mov-werks
                 berid = lv_candidate
        BINARY SEARCH.
      IF sy-subrc = 0.
        cv_berid = lv_candidate.
        RETURN.
      ENDIF.
    ENDIF.
  ENDIF.

  lv_candidate = is_mov-werks.
  READ TABLE it_mdma ASSIGNING FIELD-SYMBOL(<ls_mdma_plant>)
    WITH KEY matnr = is_mov-matnr
             werks = is_mov-werks
             berid = lv_candidate
    BINARY SEARCH.
  IF sy-subrc = 0.
    cv_berid = <ls_mdma_plant>-berid.
    RETURN.
  ENDIF.

  READ TABLE it_mdlv_by_lgort ASSIGNING FIELD-SYMBOL(<ls_mdlv_plant>)
    WITH TABLE KEY werks = is_mov-werks
                   lgort = space.
  IF sy-subrc = 0.
    lv_candidate = <ls_mdlv_plant>-berid.
    READ TABLE it_mdma TRANSPORTING NO FIELDS
      WITH KEY matnr = is_mov-matnr
               werks = is_mov-werks
               berid = lv_candidate
      BINARY SEARCH.
    IF sy-subrc = 0.
      cv_berid = lv_candidate.
    ENDIF.
  ENDIF.

ENDFORM.

*&---------------------------------------------------------------------*
*& FORM load_mdlv - Dispobereich/Lagerort-Zuordnungen laden
*&---------------------------------------------------------------------*
FORM load_mdlv CHANGING ct_mdlv TYPE tt_mdlv.

  DATA: lr_struct_descr TYPE REF TO cl_abap_structdescr,
        lr_type_descr   TYPE REF TO cl_abap_typedescr,
        lr_table_descr  TYPE REF TO cl_abap_tabledescr,
        lr_data         TYPE REF TO data,
        lv_tabname      TYPE string VALUE 'MDLV',
        ls_mdlv         TYPE ty_mdlv,
        lv_werks        TYPE werks_d.

  FIELD-SYMBOLS: <lt_raw> TYPE STANDARD TABLE,
                 <ls_raw> TYPE any,
                 <lv_any> TYPE any.

  CLEAR ct_mdlv.

  TRY.
      lr_type_descr = cl_abap_typedescr=>describe_by_name( 'MDLV' ).
      lr_struct_descr ?= lr_type_descr.
      lr_table_descr = cl_abap_tabledescr=>create(
                         p_line_type = lr_struct_descr ).
      CREATE DATA lr_data TYPE HANDLE lr_table_descr.
      ASSIGN lr_data->* TO <lt_raw>.
      IF <lt_raw> IS NOT ASSIGNED.
        RETURN.
      ENDIF.

      SELECT * FROM (lv_tabname) INTO TABLE @<lt_raw>.
    CATCH cx_root.
      RETURN.
  ENDTRY.

  LOOP AT <lt_raw> ASSIGNING <ls_raw>.
    CLEAR: ls_mdlv, lv_werks.

    ASSIGN COMPONENT 'WERKS' OF STRUCTURE <ls_raw> TO <lv_any>.
    IF sy-subrc <> 0.
      ASSIGN COMPONENT 'WERZG' OF STRUCTURE <ls_raw> TO <lv_any>.
    ENDIF.
    IF sy-subrc <> 0.
      CONTINUE.
    ENDIF.

    lv_werks = <lv_any>.
    IF lv_werks IN s_werks.
      ls_mdlv-werks = lv_werks.
    ELSE.
      CONTINUE.
    ENDIF.

    ASSIGN COMPONENT 'BERID' OF STRUCTURE <ls_raw> TO <lv_any>.
    IF sy-subrc = 0.
      ls_mdlv-berid = <lv_any>.
    ENDIF.
    ASSIGN COMPONENT 'BERTY' OF STRUCTURE <ls_raw> TO <lv_any>.
    IF sy-subrc = 0.
      ls_mdlv-berty = <lv_any>.
      SHIFT ls_mdlv-berty LEFT DELETING LEADING '0'.
    ENDIF.
    ASSIGN COMPONENT 'BERTX' OF STRUCTURE <ls_raw> TO <lv_any>.
    IF sy-subrc = 0.
      ls_mdlv-bertx = <lv_any>.
    ENDIF.
    ASSIGN COMPONENT 'ORTZG' OF STRUCTURE <ls_raw> TO <lv_any>.
    IF sy-subrc <> 0.
      ASSIGN COMPONENT 'LGORT' OF STRUCTURE <ls_raw> TO <lv_any>.
    ENDIF.
    IF sy-subrc = 0.
      ls_mdlv-lgort = <lv_any>.
    ENDIF.

    ls_mdlv-configdeprecationcode = 'Gültig'.
    APPEND ls_mdlv TO ct_mdlv.
  ENDLOOP.

  SORT ct_mdlv BY berid werks lgort.
  DELETE ADJACENT DUPLICATES FROM ct_mdlv COMPARING berid werks lgort.

ENDFORM.

*&---------------------------------------------------------------------*
*& FORM finalize_workbook - Paket als komplette Arbeitsmappe erzeugen
*&---------------------------------------------------------------------*
FORM finalize_workbook USING it_book_marc TYPE tt_marc
                             it_book_mov  TYPE tt_mov
                             it_plants    TYPE tt_plant
                             it_t006      TYPE tt_t006
                             it_t134      TYPE tt_t134
                             it_t438a     TYPE tt_t438a
                             it_t399d     TYPE tt_t399d
                             it_mdma_all  TYPE tt_mdma
                             it_mdlv_all  TYPE tt_mdlv
                             it_mbew_all  TYPE tt_mbew.

  DATA: lt_book_plants TYPE tt_plant,
        lt_book_mara   TYPE tt_mara,
        lt_book_mdma   TYPE tt_mdma,
        lt_book_mdlv   TYPE tt_mdlv,
        lt_book_mbew   TYPE tt_mbew,
        lv_xlsx        TYPE xstring.

  CLEAR gt_sheets.

  LOOP AT it_book_marc ASSIGNING FIELD-SYMBOL(<ls_book_marc>).
    " gt_plants ist nach WERKS sortiert (START-OF-SELECTION)
    READ TABLE it_plants ASSIGNING FIELD-SYMBOL(<ls_book_plant>)
      WITH KEY werks = <ls_book_marc>-werks BINARY SEARCH.
    IF sy-subrc = 0.
      APPEND <ls_book_plant> TO lt_book_plants.
    ENDIF.

    APPEND VALUE ty_mara( matnr = <ls_book_marc>-matnr
                          mtart = <ls_book_marc>-mtart
                          lvorm = space
                          meins = <ls_book_marc>-meins ) TO lt_book_mara.
  ENDLOOP.

  SORT lt_book_plants BY werks.
  DELETE ADJACENT DUPLICATES FROM lt_book_plants COMPARING werks.
  SORT lt_book_mara BY matnr.
  DELETE ADJACENT DUPLICATES FROM lt_book_mara COMPARING matnr.

  LOOP AT it_book_marc ASSIGNING <ls_book_marc>.
    LOOP AT it_mdma_all ASSIGNING FIELD-SYMBOL(<ls_book_mdma_all>)
         WHERE matnr = <ls_book_marc>-matnr
           AND werks = <ls_book_marc>-werks.
      APPEND <ls_book_mdma_all> TO lt_book_mdma.
    ENDLOOP.
  ENDLOOP.
  SORT lt_book_mdma BY matnr werks berid.
  DELETE ADJACENT DUPLICATES FROM lt_book_mdma COMPARING matnr werks berid.

  LOOP AT it_mdlv_all ASSIGNING FIELD-SYMBOL(<ls_book_mdlv_all>).
    READ TABLE lt_book_plants TRANSPORTING NO FIELDS
      WITH KEY werks = <ls_book_mdlv_all>-werks BINARY SEARCH.
    IF sy-subrc = 0.
      APPEND <ls_book_mdlv_all> TO lt_book_mdlv.
    ENDIF.
  ENDLOOP.
  SORT lt_book_mdlv BY berid werks lgort.
  DELETE ADJACENT DUPLICATES FROM lt_book_mdlv COMPARING berid werks lgort.

  LOOP AT it_book_marc ASSIGNING <ls_book_marc>.
    READ TABLE it_mbew_all ASSIGNING FIELD-SYMBOL(<ls_book_mbew>)
      WITH KEY matnr = <ls_book_marc>-matnr
               bwkey = <ls_book_marc>-bwkey
      BINARY SEARCH.
    IF sy-subrc = 0.
      APPEND <ls_book_mbew> TO lt_book_mbew.
    ENDIF.
  ENDLOOP.
  SORT lt_book_mbew BY matnr bwkey.
  DELETE ADJACENT DUPLICATES FROM lt_book_mbew COMPARING matnr bwkey.

  PERFORM build_common_sheets USING lt_book_plants
                                    it_t006
                                    it_t134
                                    it_t438a
                                    it_t399d.
  PERFORM build_master_sheets USING it_book_marc
                                    lt_book_mara
                                    lt_book_mbew.
  PERFORM build_mrp_area_sheets USING lt_book_mdma
                                      lt_book_mdlv.
  PERFORM build_matdoc_sheet USING it_book_mov.
  PERFORM reorder_sheets_like_template.
  PERFORM build_xlsx_zip CHANGING lv_xlsx.
  PERFORM publish_workbook USING lv_xlsx.

  CLEAR gt_sheets.

ENDFORM.

*&---------------------------------------------------------------------*
*& FORM publish_workbook - Arbeitsmappe puffern, streamen oder zippen
*&---------------------------------------------------------------------*
*& Server-Export : Erste Mappe wird gepuffert (Einzeldatei-Fall). Ab
*&                 der zweiten Mappe wird jede Mappe SOFORT als
*&                 nummerierte Datei geschrieben -> es liegt immer nur
*&                 eine Arbeitsmappe gleichzeitig im Speicher, kein
*&                 Sammel-ZIP.
*& Frontend       : Wie bisher - ein XLSX direkt, mehrere als ZIP
*&                 (eine Einzeldatei ist fuer den Download praktischer).
*&---------------------------------------------------------------------*
FORM publish_workbook USING iv_xlsx TYPE xstring.

  DATA: lv_entry_name TYPE string,
        lv_path       TYPE string.

  gv_workbook_count = gv_workbook_count + 1.
  PERFORM workbook_entry_name USING gv_workbook_count
                              CHANGING lv_entry_name.

  IF p_serv = 'X'.
    IF gv_workbook_count = 1.
      gv_single_xlsx = iv_xlsx.
      gv_single_name = lv_entry_name.
    ELSE.
      IF gv_workbook_count = 2.
        " Nachholen: gepufferte erste Mappe als _001 schreiben
        PERFORM server_workbook_path USING 1 CHANGING lv_path.
        PERFORM write_binary_file USING lv_path gv_single_xlsx.
        CLEAR gv_single_xlsx.
      ENDIF.
      PERFORM server_workbook_path USING gv_workbook_count
                                   CHANGING lv_path.
      PERFORM write_binary_file USING lv_path iv_xlsx.
    ENDIF.
  ELSE.
    IF gv_workbook_count = 1.
      gv_single_xlsx = iv_xlsx.
      gv_single_name = lv_entry_name.
    ELSE.
      IF go_result_zip IS INITIAL.
        CREATE OBJECT go_result_zip.
        go_result_zip->add( name = gv_single_name content = gv_single_xlsx ).
        CLEAR gv_single_xlsx.
      ENDIF.
      go_result_zip->add( name = lv_entry_name content = iv_xlsx ).
    ENDIF.
  ENDIF.

  " Fortschritt ins Job-Log; im Dialog wuerden S-Messages nur die
  " Abschlussmeldung in der Statuszeile ueberschreiben
  IF sy-batch = 'X'.
    MESSAGE |Arbeitsmappe { gv_workbook_count } erzeugt.| TYPE 'S'.
  ENDIF.

ENDFORM.

FORM workbook_entry_name USING iv_index TYPE i
                         CHANGING cv_name TYPE string.

  cv_name = |stoxion_export_{ iv_index WIDTH = 3 ALIGN = RIGHT PAD = '0' }.xlsx|.

ENDFORM.

*&---------------------------------------------------------------------*
*& FORM server_workbook_path - nummerierter Zielpfad je Arbeitsmappe
*&---------------------------------------------------------------------*
FORM server_workbook_path USING iv_index TYPE i
                          CHANGING cv_path TYPE string.

  DATA: lv_base     TYPE string,
        lv_base_len TYPE i.

  PERFORM build_workbook_filename CHANGING lv_base.  "endet immer auf .xlsx
  lv_base_len = strlen( lv_base ) - 5.
  cv_path = |{ lv_base(lv_base_len) }| &&
            |_{ iv_index WIDTH = 3 ALIGN = RIGHT PAD = '0' }.xlsx|.

ENDFORM.

*&---------------------------------------------------------------------*
*& FORM write_export_result - Restpuffer schreiben, Zielnamen setzen
*&---------------------------------------------------------------------*
FORM write_export_result.

  DATA: lv_result_file TYPE string,
        lv_last_file   TYPE string,
        lv_zip         TYPE xstring.

  IF gv_workbook_count = 0.
    MESSAGE 'Keine Arbeitsmappe fuer XLSX-Erzeugung vorhanden.' TYPE 'E'.
  ENDIF.

  IF p_serv = 'X'.
    IF gv_workbook_count = 1.
      " Einzelfall: gepufferte Mappe unter dem einfachen Zielnamen
      PERFORM build_workbook_filename CHANGING gv_xlsx_file.
      PERFORM write_binary_file USING gv_xlsx_file gv_single_xlsx.
      CLEAR gv_single_xlsx.
    ELSE.
      " Mappen wurden bereits einzeln geschrieben -> nur Anzeigename
      PERFORM server_workbook_path USING 1 CHANGING gv_xlsx_file.
      PERFORM server_workbook_path USING gv_workbook_count
                                   CHANGING lv_last_file.
      gv_xlsx_file = |{ gv_xlsx_file } bis { lv_last_file }|.
    ENDIF.
  ELSE.
    PERFORM build_workbook_filename CHANGING gv_xlsx_file.
    IF gv_workbook_count = 1.
      PERFORM write_binary_file USING gv_xlsx_file gv_single_xlsx.
    ELSE.
      IF go_result_zip IS INITIAL.
        MESSAGE 'ZIP-Erzeugung nicht initialisiert.' TYPE 'E'.
      ENDIF.
      lv_zip = go_result_zip->save( ).
      PERFORM build_zip_filename USING gv_xlsx_file
                                 CHANGING lv_result_file.
      gv_xlsx_file = lv_result_file.
      PERFORM write_binary_file USING gv_xlsx_file lv_zip.
    ENDIF.
  ENDIF.

ENDFORM.

FORM build_zip_filename USING iv_xlsx_file TYPE string
                        CHANGING cv_zip_file TYPE string.

  DATA: lv_len      TYPE i,
        lv_base_len TYPE i.

  cv_zip_file = iv_xlsx_file.
  lv_len = strlen( cv_zip_file ).
  IF lv_len > 5.
    lv_base_len = lv_len - 5.
    cv_zip_file = cv_zip_file(lv_base_len) && '.zip'.
  ELSE.
    cv_zip_file = cv_zip_file && '.zip'.
  ENDIF.

ENDFORM.

FORM write_binary_file USING iv_file    TYPE string
                             iv_content TYPE xstring.

  DATA lt_bin TYPE solix_tab.

  IF p_serv = 'X'.
    CALL FUNCTION 'AUTHORITY_CHECK_DATASET'
      EXPORTING
        activity         = sabc_act_write
        filename         = iv_file
      EXCEPTIONS
        no_authority     = 1
        activity_unknown = 2
        OTHERS           = 3.
    IF sy-subrc <> 0.
      MESSAGE |Keine Schreibberechtigung fuer { iv_file }.| TYPE 'E'.
    ENDIF.

    OPEN DATASET iv_file FOR OUTPUT IN BINARY MODE.
    IF sy-subrc <> 0.
      MESSAGE |Datei { iv_file } nicht oeffenbar.| TYPE 'E'.
    ENDIF.
    TRANSFER iv_content TO iv_file.
    CLOSE DATASET iv_file.
  ELSE.
    lt_bin = cl_bcs_convert=>xstring_to_solix( iv_xstring = iv_content ).
    cl_gui_frontend_services=>gui_download(
      EXPORTING
        bin_filesize = xstrlen( iv_content )
        filename     = iv_file
        filetype     = 'BIN'
      CHANGING
        data_tab     = lt_bin
      EXCEPTIONS
        OTHERS       = 1 ).
    IF sy-subrc <> 0.
      MESSAGE |Download { iv_file } fehlgeschlagen (rc={ sy-subrc }).|
        TYPE 'E'.
    ENDIF.
  ENDIF.

ENDFORM.

*&---------------------------------------------------------------------*
*& Sheet-Writer: Reiter sammeln, am Ende als eine XLSX schreiben
*&---------------------------------------------------------------------*
FORM sheet_open USING iv_name TYPE string.

  gv_sh_name = iv_name.
  gv_sh_isopen = abap_true.
  CLEAR gt_sh_buffer.

ENDFORM.

FORM sheet_put USING it_lines TYPE STANDARD TABLE.

  APPEND LINES OF it_lines TO gt_sh_buffer.

ENDFORM.

FORM sheet_close.

  IF gv_sh_isopen = abap_false.
    RETURN.
  ENDIF.

  APPEND VALUE ty_sheet( name  = gv_sh_name
                         lines = gt_sh_buffer ) TO gt_sheets.
  CLEAR gt_sh_buffer.
  CLEAR gv_sh_name.
  gv_sh_isopen = abap_false.

ENDFORM.

*&---------------------------------------------------------------------*
*& FORM write_sheet - komplette Reiter-Datei in einem Zug
*&---------------------------------------------------------------------*
FORM write_sheet USING iv_name  TYPE string
                       it_lines TYPE STANDARD TABLE.

  PERFORM sheet_open  USING iv_name.
  PERFORM sheet_put   USING it_lines.
  PERFORM sheet_close.

ENDFORM.

*&---------------------------------------------------------------------*
*& FORM validate_target_path - Zielparameter vor dem Export pruefen
*&---------------------------------------------------------------------*
FORM validate_target_path.

  IF p_path IS INITIAL.
    MESSAGE 'Bitte Zielordner, XLSX- oder ZIP-Datei angeben.' TYPE 'E'.
  ENDIF.

  IF p_serv = 'X' AND ( p_path CS ':\' OR p_path CS ':/' OR
                        p_path CP '\\*' ).
    MESSAGE 'Fuer Server-Export bitte AL11-/Unix-Pfad angeben, ' &&
            'z.B. /tmp/stoxion/stoxion_export.xlsx (kein lokales ' &&
            'Laufwerk, keine UNC-Freigabe).' TYPE 'E'.
  ENDIF.

ENDFORM.

*&---------------------------------------------------------------------*
*& FORM validate_max_rows - Zeilenlimit je Arbeitsmappe pruefen
*&---------------------------------------------------------------------*
FORM validate_max_rows.

  IF p_maxrow < 1000.
    MESSAGE 'Max. Zeilen je Arbeitsmappe: mindestens 1.000 angeben.'
      TYPE 'E'.
  ENDIF.
  IF p_maxrow > gc_excel_max_rows.
    MESSAGE |Max. Zeilen je Arbeitsmappe: hoechstens | &&
            |{ gc_excel_max_rows } (Excel-Limit je Sheet).| TYPE 'E'.
  ENDIF.

ENDFORM.

*&---------------------------------------------------------------------*
*& FORM build_workbook_filename - Zielpfad + XLSX-Dateiname
*&---------------------------------------------------------------------*
FORM build_workbook_filename CHANGING cv_file TYPE string.

  DATA(lv_sep) = COND string( WHEN p_front = 'X' THEN '\' ELSE '/' ).

  cv_file = p_path.
  IF cv_file IS INITIAL.
    MESSAGE 'Bitte Zielordner, XLSX- oder ZIP-Datei angeben.' TYPE 'E'.
  ENDIF.

  DATA(lv_lower) = to_lower( val = cv_file ).
  IF lv_lower CP '*.xlsx'.
    RETURN.
  ENDIF.
  IF lv_lower CP '*.zip'.
    DATA(lv_zip_base_len) = strlen( cv_file ) - 4.
    cv_file = cv_file(lv_zip_base_len) && '.xlsx'.
    RETURN.
  ENDIF.

  " Beide Separatoren pruefen: 'C:/stoxion/' am Frontend soll nicht
  " zu 'C:/stoxion/\stoxion_export.xlsx' fuehren
  DATA(lv_len) = strlen( cv_file ) - 1.
  IF cv_file+lv_len(1) <> '\' AND cv_file+lv_len(1) <> '/'.
    cv_file = cv_file && lv_sep.
  ENDIF.
  cv_file = |{ cv_file }stoxion_export.xlsx|.

ENDFORM.

*&---------------------------------------------------------------------*
*& FORM build_xlsx_zip - minimale OpenXML-Arbeitsmappe bauen
*&---------------------------------------------------------------------*
FORM build_xlsx_zip CHANGING cv_xlsx TYPE xstring.

  DATA: lo_zip         TYPE REF TO cl_abap_zip,
        lv_content     TYPE string,
        lv_workbook    TYPE string,
        lv_rels        TYPE string,
        lv_root_rels   TYPE string,
        lv_sheet_xml   TYPE string,
        lv_sheet_name  TYPE string,
        lv_part_name   TYPE string,
        lv_sheet_no    TYPE i,
        lv_sheet_no_c  TYPE string.

  CREATE OBJECT lo_zip.

  lv_content =
    |<?xml version="1.0" encoding="UTF-8" standalone="yes"?>| &&
    |<Types xmlns="http://schemas.openxmlformats.org/package/2006/content-types">| &&
    |<Default Extension="rels" ContentType="application/vnd.openxmlformats-package.relationships+xml"/>| &&
    |<Default Extension="xml" ContentType="application/xml"/>| &&
    |<Override PartName="/xl/workbook.xml" ContentType="application/vnd.openxmlformats-officedocument.spreadsheetml.sheet.main+xml"/>|.

  lv_workbook =
    |<?xml version="1.0" encoding="UTF-8" standalone="yes"?>| &&
    |<workbook xmlns="http://schemas.openxmlformats.org/spreadsheetml/2006/main" | &&
    |xmlns:r="http://schemas.openxmlformats.org/officeDocument/2006/relationships"><sheets>|.

  lv_rels =
    |<?xml version="1.0" encoding="UTF-8" standalone="yes"?>| &&
    |<Relationships xmlns="http://schemas.openxmlformats.org/package/2006/relationships">|.

  lv_sheet_no = 0.
  LOOP AT gt_sheets ASSIGNING FIELD-SYMBOL(<ls_sheet>).
    lv_sheet_no = lv_sheet_no + 1.
    lv_sheet_no_c = lv_sheet_no.
    CONDENSE lv_sheet_no_c NO-GAPS.
    PERFORM xml_escape USING <ls_sheet>-name CHANGING lv_sheet_name.

    lv_content = lv_content &&
      |<Override PartName="/xl/worksheets/sheet{ lv_sheet_no_c }.xml" | &&
      |ContentType="application/vnd.openxmlformats-officedocument.spreadsheetml.worksheet+xml"/>|.

    lv_workbook = lv_workbook &&
      |<sheet name="{ lv_sheet_name }" sheetId="{ lv_sheet_no_c }" | &&
      |r:id="rId{ lv_sheet_no_c }"/>|.

    lv_rels = lv_rels &&
      |<Relationship Id="rId{ lv_sheet_no_c }" | &&
      |Type="http://schemas.openxmlformats.org/officeDocument/2006/relationships/worksheet" | &&
      |Target="worksheets/sheet{ lv_sheet_no_c }.xml"/>|.

    PERFORM build_worksheet_xml USING <ls_sheet>-lines CHANGING lv_sheet_xml.
    lv_part_name = |xl/worksheets/sheet{ lv_sheet_no_c }.xml|.
    PERFORM add_zip_string USING lo_zip lv_part_name lv_sheet_xml.
  ENDLOOP.

  lv_content = lv_content && |</Types>|.
  lv_workbook = lv_workbook && |</sheets></workbook>|.
  lv_rels = lv_rels && |</Relationships>|.

  PERFORM add_zip_string USING lo_zip '[Content_Types].xml' lv_content.
  PERFORM add_zip_string USING lo_zip 'xl/workbook.xml' lv_workbook.
  PERFORM add_zip_string USING lo_zip 'xl/_rels/workbook.xml.rels' lv_rels.
  lv_root_rels =
    |<?xml version="1.0" encoding="UTF-8" standalone="yes"?>| &&
    |<Relationships xmlns="http://schemas.openxmlformats.org/package/2006/relationships">| &&
    |<Relationship Id="rId1" Type="http://schemas.openxmlformats.org/officeDocument/2006/relationships/officeDocument" Target="xl/workbook.xml"/>| &&
    |</Relationships>|.
  PERFORM add_zip_string USING lo_zip '_rels/.rels' lv_root_rels.

  cv_xlsx = lo_zip->save( ).

ENDFORM.

FORM build_worksheet_xml USING it_lines TYPE tt_string
                      CHANGING cv_xml   TYPE string.

  DATA: lt_cells     TYPE STANDARD TABLE OF string,
        lv_row       TYPE i,
        lv_col       TYPE i,
        lv_max_col   TYPE i,
        lv_col_name  TYPE string,
        lv_last_col  TYPE string,
        lv_cell_ref  TYPE string,
        lv_cell_xml  TYPE string,
        lv_value     TYPE string,
        lv_row_c     TYPE string,
        lv_last_row_c TYPE string.

  cv_xml =
    |<?xml version="1.0" encoding="UTF-8" standalone="yes"?>| &&
    |<worksheet xmlns="http://schemas.openxmlformats.org/spreadsheetml/2006/main">| &&
    |<sheetViews><sheetView workbookViewId="0">| &&
    |<pane ySplit="2" topLeftCell="A3" activePane="bottomLeft" state="frozen"/>| &&
    |</sheetView></sheetViews>| &&
    |<sheetData>|.

  lv_row = 0.
  lv_max_col = 0.
  LOOP AT it_lines ASSIGNING FIELD-SYMBOL(<lv_line>).
    lv_row = lv_row + 1.
    lv_row_c = lv_row.
    CONDENSE lv_row_c NO-GAPS.
    SPLIT <lv_line> AT gc_tab INTO TABLE lt_cells.
    IF lv_row = 2.
      lv_max_col = lines( lt_cells ).
    ELSEIF lv_max_col = 0 OR lines( lt_cells ) > lv_max_col.
      lv_max_col = lines( lt_cells ).
    ENDIF.
    cv_xml = cv_xml && |<row r="{ lv_row_c }">|.

    lv_col = 0.
    LOOP AT lt_cells ASSIGNING FIELD-SYMBOL(<lv_cell>).
      lv_col = lv_col + 1.
      PERFORM column_name USING lv_col CHANGING lv_col_name.
      lv_cell_ref = |{ lv_col_name }{ lv_row_c }|.
      PERFORM xml_escape USING <lv_cell> CHANGING lv_value.
      lv_cell_xml =
        |<c r="{ lv_cell_ref }" t="inlineStr"><is><t>{ lv_value }</t></is></c>|.
      cv_xml = cv_xml && lv_cell_xml.
    ENDLOOP.

    cv_xml = cv_xml && |</row>|.
  ENDLOOP.

  cv_xml = cv_xml && |</sheetData>|.
  IF lv_max_col > 0.
    PERFORM column_name USING lv_max_col CHANGING lv_last_col.
    lv_last_row_c = COND string( WHEN lv_row > 2 THEN lv_row ELSE 2 ).
    CONDENSE lv_last_row_c NO-GAPS.
    cv_xml = cv_xml &&
      |<autoFilter ref="A2:{ lv_last_col }{ lv_last_row_c }"/>|.
  ENDIF.
  cv_xml = cv_xml && |</worksheet>|.

ENDFORM.

FORM column_name USING iv_col TYPE i CHANGING cv_name TYPE string.

  DATA: lv_col TYPE i,
        lv_mod TYPE i,
        lv_chr TYPE c LENGTH 1.

  CLEAR cv_name.
  lv_col = iv_col.
  WHILE lv_col > 0.
    lv_mod = ( lv_col - 1 ) MOD 26.
    lv_chr = sy-abcde+lv_mod(1).
    cv_name = lv_chr && cv_name.
    lv_col = ( lv_col - 1 ) DIV 26.
  ENDWHILE.

ENDFORM.

FORM xml_escape USING iv_value TYPE string CHANGING cv_value TYPE string.

  cv_value = iv_value.
  REPLACE ALL OCCURRENCES OF '&' IN cv_value WITH '&amp;'.
  REPLACE ALL OCCURRENCES OF '<' IN cv_value WITH '&lt;'.
  REPLACE ALL OCCURRENCES OF '>' IN cv_value WITH '&gt;'.
  REPLACE ALL OCCURRENCES OF '"' IN cv_value WITH '&quot;'.
  REPLACE ALL OCCURRENCES OF '''' IN cv_value WITH '&apos;'.

ENDFORM.

FORM add_zip_string USING io_zip  TYPE REF TO cl_abap_zip
                          iv_name TYPE string
                          iv_text TYPE string.

  DATA: lo_conv    TYPE REF TO cl_abap_conv_out_ce,
        lv_content TYPE xstring.

  lo_conv = cl_abap_conv_out_ce=>create( encoding = 'UTF-8' ).
  lo_conv->write( data = iv_text ).
  lv_content = lo_conv->get_buffer( ).
  io_zip->add( name = iv_name content = lv_content ).

ENDFORM.

*&---------------------------------------------------------------------*
*& FORM schedule_job - aktuellen Selektionsstand als Job einplanen
*&---------------------------------------------------------------------*
FORM schedule_job.

  DATA: lt_par      TYPE TABLE OF rsparams,
        lv_jobname  TYPE tbtcjob-jobname,
        lv_jobcount TYPE tbtcjob-jobcount.

  " Aktuelle Selektionswerte einsammeln
  CALL FUNCTION 'RS_REFRESH_FROM_SELECTOPTIONS'
    EXPORTING
      curr_report     = sy-repid
    TABLES
      selection_table = lt_par
    EXCEPTIONS
      OTHERS          = 1.
  IF sy-subrc <> 0.
    MESSAGE 'Selektionswerte konnten nicht gelesen werden.' TYPE 'E'.
  ENDIF.

  " Im Job zwingend Server-Export
  LOOP AT lt_par ASSIGNING FIELD-SYMBOL(<ls_par>).
    CASE <ls_par>-selname.
      WHEN 'P_FRONT'. <ls_par>-low = space.
      WHEN 'P_SERV'.  <ls_par>-low = 'X'.
    ENDCASE.
  ENDLOOP.

  " Plausibilitaet: Pfad muss ein Server-Pfad sein
  IF p_path CS ':\' OR p_path CS ':/' OR p_path CP '\\*'.
    MESSAGE 'Fuer den Job einen Server-Pfad (AL11) angeben, ' &&
            'z.B. /tmp/stoxion/ - der aktuelle Pfad zeigt auf ein ' &&
            'lokales Laufwerk oder eine Netzwerkfreigabe.' TYPE 'E'.
  ENDIF.

  lv_jobname = |ZSTOXION_{ sy-datum }_{ sy-uzeit }|.

  CALL FUNCTION 'JOB_OPEN'
    EXPORTING
      jobname          = lv_jobname
    IMPORTING
      jobcount         = lv_jobcount
    EXCEPTIONS
      cant_create_job  = 1
      invalid_job_data = 2
      jobname_missing  = 3
      OTHERS           = 4.
  IF sy-subrc <> 0.
    MESSAGE 'Job konnte nicht angelegt werden.' TYPE 'E'.
  ENDIF.

  SUBMIT (sy-repid)
    WITH SELECTION-TABLE lt_par
    VIA JOB lv_jobname NUMBER lv_jobcount
    AND RETURN.

  CALL FUNCTION 'JOB_CLOSE'
    EXPORTING
      jobname              = lv_jobname
      jobcount             = lv_jobcount
      strtimmed            = 'X'                "Sofortstart
    EXCEPTIONS
      cant_start_immediate = 1
      invalid_startdate    = 2
      jobname_missing      = 3
      job_close_failed     = 4
      job_nosteps          = 5
      job_notex            = 6
      lock_failed          = 7
      OTHERS               = 8.
  IF sy-subrc <> 0.
    MESSAGE 'Job konnte nicht freigegeben werden (SM37 pruefen).'
      TYPE 'E'.
  ENDIF.

  MESSAGE |Job { lv_jobname } eingeplant (Sofortstart). | &&
          |Ueberwachung in SM37, Dateien danach in AL11.| TYPE 'S'.

ENDFORM.

CLASS ltcl_demand_history DEFINITION FINAL FOR TESTING
  DURATION SHORT
  RISK LEVEL HARMLESS.
  PRIVATE SECTION.
    CONSTANTS: c_window_end     TYPE sy-datum VALUE '20260701',
               c_start_max      TYPE sy-datum VALUE '20230701',
               c_start_fallback TYPE sy-datum VALUE '20240701'.

    METHODS third_year_extends_window FOR TESTING.
    METHODS no_third_year_keeps_24 FOR TESTING.
    METHODS current_month_is_excluded FOR TESTING.
    METHODS raw_values_not_averaged FOR TESTING.
ENDCLASS.

CLASS ltcl_demand_history IMPLEMENTATION.

  METHOD third_year_extends_window.
    DATA: lt_mov  TYPE tt_mov,
          lt_marc TYPE tt_marc,
          lt_mdma TYPE tt_mdma,
          lt_mdlv TYPE tt_mdlv.

    APPEND VALUE ty_marc( matnr = 'MAT-156'
                          werks = '1000'
                          diber = 'X' ) TO lt_marc.
    APPEND VALUE ty_mdma( matnr = 'MAT-156'
                          werks = '1000'
                          berid = 'MRPAREA1' ) TO lt_mdma.
    APPEND VALUE ty_mdlv( berid = 'MRPAREA1'
                          werks = '1000'
                          lgort = '0001' ) TO lt_mdlv.
    APPEND VALUE ty_mov( matnr = 'MAT-156'
                         werks = '1000'
                         lgort = '0001'
                         budat = '20230715'
                         menge = '3' ) TO lt_mov.
    APPEND VALUE ty_mov( matnr = 'MAT-156'
                         werks = '1000'
                         lgort = '0001'
                         budat = '20260615'
                         menge = '4' ) TO lt_mov.

    SORT lt_marc BY matnr werks.
    SORT lt_mdma BY matnr werks berid.
    PERFORM filter_matdoc_history_window
      USING c_window_end
            c_start_max
            c_start_fallback
            lt_marc
            lt_mdma
            lt_mdlv
      CHANGING lt_mov.

    cl_abap_unit_assert=>assert_equals(
      exp = 2
      act = lines( lt_mov )
      msg = 'Buchung im Monat 25-36 muss das 36-Monats-Fenster aktivieren.' ).
    READ TABLE lt_mov ASSIGNING FIELD-SYMBOL(<ls_oldest>) INDEX 1.
    cl_abap_unit_assert=>assert_equals(
      exp = 0
      act = sy-subrc ).
    cl_abap_unit_assert=>assert_equals(
      exp = '20230715'
      act = <ls_oldest>-budat
      msg = 'Die Drittjahresbuchung bleibt im Rohdatenexport erhalten.' ).
  ENDMETHOD.

  METHOD no_third_year_keeps_24.
    DATA: lt_mov  TYPE tt_mov,
          lt_marc TYPE tt_marc,
          lt_mdma TYPE tt_mdma,
          lt_mdlv TYPE tt_mdlv.

    APPEND VALUE ty_marc( matnr = 'MAT-104'
                          werks = '1000' ) TO lt_marc.
    APPEND VALUE ty_mov( matnr = 'MAT-104'
                         werks = '1000'
                         budat = '20240701'
                         menge = '5' ) TO lt_mov.
    APPEND VALUE ty_mov( matnr = 'MAT-104'
                         werks = '1000'
                         budat = '20260630'
                         menge = '6' ) TO lt_mov.

    SORT lt_marc BY matnr werks.
    SORT lt_mdma BY matnr werks berid.
    PERFORM filter_matdoc_history_window
      USING c_window_end
            c_start_max
            c_start_fallback
            lt_marc
            lt_mdma
            lt_mdlv
      CHANGING lt_mov.

    cl_abap_unit_assert=>assert_equals(
      exp = 2
      act = lines( lt_mov )
      msg = 'Ohne Drittjahresbuchung werden keine kuenstlichen Nullwochen ergaenzt.' ).
    LOOP AT lt_mov ASSIGNING FIELD-SYMBOL(<ls_mov>).
      cl_abap_unit_assert=>assert_true(
        act = xsdbool( <ls_mov>-budat >= c_start_fallback )
        msg = 'Ohne Drittjahresbuchung bleibt die effektive Rohhistorie bei 24 Monaten.' ).
    ENDLOOP.
  ENDMETHOD.

  METHOD current_month_is_excluded.
    DATA: lt_mov  TYPE tt_mov,
          lt_marc TYPE tt_marc,
          lt_mdma TYPE tt_mdma,
          lt_mdlv TYPE tt_mdlv.

    APPEND VALUE ty_marc( matnr = 'MAT-CUR'
                          werks = '1000' ) TO lt_marc.
    APPEND VALUE ty_mov( matnr = 'MAT-CUR'
                         werks = '1000'
                         budat = '20260630'
                         menge = '1' ) TO lt_mov.
    APPEND VALUE ty_mov( matnr = 'MAT-CUR'
                         werks = '1000'
                         budat = '20260701'
                         menge = '9' ) TO lt_mov.

    SORT lt_marc BY matnr werks.
    SORT lt_mdma BY matnr werks berid.
    PERFORM filter_matdoc_history_window
      USING c_window_end
            c_start_max
            c_start_fallback
            lt_marc
            lt_mdma
            lt_mdlv
      CHANGING lt_mov.

    cl_abap_unit_assert=>assert_equals(
      exp = 1
      act = lines( lt_mov )
      msg = 'Der laufende Monat darf nicht im MATDOC-Fenster bleiben.' ).
    READ TABLE lt_mov ASSIGNING FIELD-SYMBOL(<ls_mov>) INDEX 1.
    cl_abap_unit_assert=>assert_equals(
      exp = '20260630'
      act = <ls_mov>-budat ).
  ENDMETHOD.

  METHOD raw_values_not_averaged.
    DATA: lt_mov  TYPE tt_mov,
          lt_marc TYPE tt_marc,
          lt_mdma TYPE tt_mdma,
          lt_mdlv TYPE tt_mdlv,
          lv_sum  TYPE menge_d,
          lv_exp  TYPE menge_d.

    APPEND VALUE ty_marc( matnr = 'MAT-MC'
                          werks = '1000' ) TO lt_marc.
    APPEND VALUE ty_mov( matnr = 'MAT-MC'
                         werks = '1000'
                         budat = '20250106'
                         menge = '2' ) TO lt_mov.
    APPEND VALUE ty_mov( matnr = 'MAT-MC'
                         werks = '1000'
                         budat = '20250108'
                         menge = '8' ) TO lt_mov.

    SORT lt_marc BY matnr werks.
    SORT lt_mdma BY matnr werks berid.
    PERFORM filter_matdoc_history_window
      USING c_window_end
            c_start_max
            c_start_fallback
            lt_marc
            lt_mdma
            lt_mdlv
      CHANGING lt_mov.

    cl_abap_unit_assert=>assert_equals(
      exp = 2
      act = lines( lt_mov )
      msg = 'Monte-Carlo-Eingangsdaten bleiben echte Buchungen, keine Kalenderwochen-Mittelwerte.' ).

    LOOP AT lt_mov ASSIGNING FIELD-SYMBOL(<ls_week_mov>).
      lv_sum = lv_sum + <ls_week_mov>-menge.
    ENDLOOP.
    lv_exp = '10'.
    cl_abap_unit_assert=>assert_equals(
      exp = lv_exp
      act = lv_sum
      msg = 'Rohmengen muessen unveraendert summierbar bleiben.' ).
  ENDMETHOD.

ENDCLASS.
